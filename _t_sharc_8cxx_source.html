95void TSharc::AddFragment(const std::shared_ptr<const TFragment>& frag, TChannel* chan)
96{
97 if(frag == nullptr || chan == nullptr) {

116void TSharc::BuildHits()
117{
118 std::vector<TFragment>::iterator front;
119 std::vector<TFragment>::iterator back;
120 std::vector<TFragment>::iterator pad;
121
122 for(front = fFrontFragments.begin(); front != fFrontFragments.end();) {
123 bool front_used = false;
124 bool back_used = false;
125 for(back = fBackFragments.begin(); back != fBackFragments.end(); back++) {
126 if(front->GetDetector() == back->GetDetector()) {
127 front_used = true;
128 back_used = true;
129 break;
130 }
131 }
132 if(front_used && back_used) {
133 TSharcHit* hit = new TSharcHit;
134 hit->SetFront(*front);
135 hit->SetBack(*back);
136 fHits.push_back(hit);
137 front = fFrontFragments.erase(front);
138 back = fBackFragments.erase(back);
139 } else {
140 front++;
141 }
142 }
143
144 for(auto& fSharcHit : fHits) {
145 for(pad = fPadFragments.begin(); pad != fPadFragments.end(); pad++) {
146 if(fSharcHit->GetDetector() == pad->GetDetector()) {
147 static_cast<TSharcHit*>(fSharcHit)->SetPad(*pad);
148 pad = fPadFragments.erase(pad);
149 break;
150 }
151 }
152 }
153}

205TVector3 TSharc::GetPosition(int detector, int frontstrip, int backstrip, double X, double Y, double Z)
206{
207 int FrontDet = detector;
208 int FrontStr = frontstrip;
209 // int BackDet = detector;
210 int BackStr = backstrip;
211 int nrots = 0; // allows us to rotate into correct position
212
213 TVector3 position;
214 TVector3 position_offset;
215 position_offset.SetXYZ(X, Y, Z);
216
217 if(FrontDet >= 5 && FrontDet <= 8) { // forward box
218 nrots = FrontDet - 4; // edited to make box 5 on the ceiling. assuming rotaing ccw around the +z axis!!
219 double x = fXposDB; // ?? x stays the same. first detector is aways defined in the y-z plane.
220 double y = -(fYminDB + (FrontStr + 0.5) * fStripFPitch); // [(-36.0) - (+36.0)] // ?? add minus sign,
221 // reversve the order of the strips on the ds section.
222 double z = fZminDB + (BackStr + 0.5) * fStripBPitch; // [(+9.0) - (+57.0)]
223 position.SetXYZ(x, y, z);
224 } else if(FrontDet >= 9 && FrontDet <= 12) { // backward box
225 nrots = FrontDet - 8; // edited to make box 5 on the ceiling. assuming rotaing ccw around the +z axis!!
226 double x = fXposUB;
227 double y = fYminUB + (FrontStr + 0.5) * fStripFPitch; // [(-36.0) - (+36.0)]
228 double z = fZminUB - (BackStr + 0.5) * fStripBPitch; // [(-5.0) - (-53.0)]
229 position.SetXYZ(x, y, z);
230 } else if(FrontDet >= 13) { // backward (upstream) QQQ
231 nrots = FrontDet - 13;
232 double z = fZposUQ;
233 double rho = fRmaxUQ - (FrontStr + 0.5) * fRingPitch; // [(+9.0) - (+41.0)]
234 double phi = (fPminUQ + (BackStr + 0.5) * fSegmentPitch) * TMath::Pi() / 180.0; // [(+2.0) - (+83.6)]
235 position.SetXYZ(rho * TMath::Sin(phi), rho * TMath::Cos(phi), z);
236 } else if(FrontDet <= 4) { // forward (downstream) QQQ
237 nrots = FrontDet - 1;
238 double z = fZposDQ;
239 double rho = fRmaxDQ - (FrontStr + 0.5) * fRingPitch; // [(+9.0) - (+41.0)]
240 double phi = (fPminDQ + (BackStr + 0.5) * fSegmentPitch) * TMath::Pi() / 180.0; // [(+6.4) - (+88.0)]
241 position.SetXYZ(rho * TMath::Sin(phi), rho * TMath::Cos(phi), z);
242 }
243
244 position.RotateZ(TMath::Pi() * nrots / 2);
245 return (position + position_offset);
246}

248double TSharc::GetDetectorThickness(TSharcHit& hit, double dist)
249{
250 static double fDetectorThickness[16] = {998., 998., 998., 1001., 141., 142., 133., 143.,
251 999., 1001., 1001., 1002., 390., 390., 383., 385.};
252 if(dist < 0.0) {
253 dist = fDetectorThickness[hit.GetDetector()];
254 }
255
256 double phi_90 = fmod(std::fabs(hit.GetPosition().Phi()), TMath::Pi() / 2);
257 double phi_45 = phi_90;
258 if(phi_90 > (TMath::Pi() / 4.)) {
259 phi_45 = TMath::Pi() / 2 - phi_90;
260 }
261
262 if(hit.GetDetector() >= 5 && hit.GetDetector() <= 12) {
263 return dist / (TMath::Sin(hit.GetPosition().Theta()) * TMath::Cos(phi_45));
264 }
265 return std::fabs(dist / (TMath::Cos(hit.GetPosition().Theta())));
266}
