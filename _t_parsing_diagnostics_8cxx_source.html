32void TParsingDiagnosticsData::Update(const std::shared_ptr<const TFragment>& frag)
33{
34 UInt_t channelId = frag->GetChannelId();
35 long timeStamp = frag->GetTimeStampNs();
36
37 // update minimum and maximum channel id if necessary
38 if(channelId < fMinChannelId) {
39 fMinChannelId = channelId;
40 }
41 if(channelId > fMaxChannelId) {
42 fMaxChannelId = channelId;
43 }
44
45 ++fNumberOfHits;
46
47 // increment the dead time and set per channel min/max timestamps
48 fDeadTime += frag->GetDeadTime();
49 if(timeStamp < fMinTimeStamp) {

138void TParsingDiagnostics::GoodFragment(const std::shared_ptr<const TFragment>& frag)
139{
140 /// increment the counter of good fragments for this detector type and check if any trigger ids have been lost
141 fNumberOfGoodFragments[frag->GetDetectorType()]++;
142
143 UInt_t channelAddress = frag->GetAddress();
144 if(fChannelAddressData.find(channelAddress) == fChannelAddressData.end()) {
145 fChannelAddressData[channelAddress] = TParsingDiagnosticsData(frag);
146 } else {
147 fChannelAddressData[channelAddress].Update(frag);
148 }
149
150 // check if this is a new minimum/maximum network packet id
151 if(frag->GetNetworkPacketNumber() > 0) {
152 ++fNumberOfNetworkPackets;
153 if(frag->GetNetworkPacketNumber() < fMinNetworkPacketNumber) {
154 fMinNetworkPacketNumber = frag->GetNetworkPacketNumber();
155 }
156 if(frag->GetNetworkPacketNumber() > fMaxNetworkPacketNumber) {
157 fMaxNetworkPacketNumber = frag->GetNetworkPacketNumber();
158 }
159 }
160
161 if(fMinDaqTimeStamp == 0 || frag->GetDaqTimeStamp() < fMinDaqTimeStamp) {
162 fMinDaqTimeStamp = frag->GetDaqTimeStamp();
163 }
164 if(fMaxDaqTimeStamp == 0 || frag->GetDaqTimeStamp() > fMaxDaqTimeStamp) {
165 fMaxDaqTimeStamp = frag->GetDaqTimeStamp();

178void TParsingDiagnostics::Draw(Option_t* opt)
179{
180 UInt_t minChannel = fChannelAddressData.begin()->first;
181 UInt_t maxChannel = fChannelAddressData.begin()->first;
182
183 for(auto it : fChannelAddressData) {
184 if(it.first < minChannel) minChannel = it.first;
185 if(it.first > maxChannel) maxChannel = it.first;
186 }
187
188 // check that the histogram (if it already exists) has the right number of bins
189 if(fIdHist != nullptr && fIdHist->GetNbinsX() != static_cast<Int_t>(maxChannel - minChannel + 1)) {
190 delete fIdHist;
191 fIdHist = nullptr;
192 }
193 if(fIdHist == nullptr) {
194 fIdHist = new TH1F("IdHist", "Event survival;channel number;survival rate [%]", maxChannel - minChannel + 1,
195 minChannel, maxChannel + 1);
196 } else {
197 // the histogram already had the right number of bins, but to be save we set the range
198 fIdHist->SetAxisRange(minChannel, maxChannel + 1);
199 }
200
201 for(auto it : fChannelAddressData) {
202 if(it.second.MinChannelId() != 0 || it.second.MinChannelId() != it.second.MaxChannelId()) {
203 fIdHist->SetBinContent(fIdHist->GetXaxis()->FindBin(it.first),
204 (100. * it.second.NumberOfHits()) / (it.second.MaxChannelId() - it.second.MinChannelId() + 1.));
205 }
206 }
207
208 fIdHist->Draw(opt);
209}
