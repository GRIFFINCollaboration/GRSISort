177void TChannel::AddChannel(TChannel* chan, Option_t* opt)
178{
179 /// Add a TChannel to fChannelMap. If the TChannel doesn't exist, create a new TChannel and add that the fChannelMap.
180 /// Options:
181 /// "overwrite" - The TChannel in the fChannelMap at the same address is overwritten.
182 /// If this option is not specified, an Error is returned if the TChannel already
183 /// exists in the fChannelMap.
184 /// "save" - The temporary channel is not deleted after being placed in the map.
185 if(chan == nullptr) {
186 return;
187 }
188 if(fChannelMap->count(chan->GetAddress()) == 1) { // if this channel exists
189 if(strcmp(opt, "overwrite") == 0) {
190 TChannel* oldchan = GetChannel(chan->GetAddress());
191 int oldNumber = oldchan->GetNumber();
192 oldchan->OverWriteChannel(chan);
193 // Need to also update the channel number map RD
194 if(oldNumber != oldchan->GetNumber()) {
195 // channel number has changed so we need to delete the old one and insert the new one
196 fChannelNumberMap->erase(oldNumber);
197 if((oldchan->GetNumber() != 0) && (fChannelNumberMap->count(oldchan->GetNumber()) == 0)) {
198 fChannelNumberMap->insert(std::make_pair(oldchan->GetNumber(), oldchan));
199 }
200 }
201 return;
202 }
203 std::cout<<"Trying to add a channel that already exists!"<<std::endl;
204 return;
205 }
206 if((chan->GetAddress() & 0x00ffffff) == 0x00ffffff) {
207 // this is the default tigress value for i am not there.
208 // we should not imclude it in the map.
209 delete chan;
210 } else {
211 // We need to update the channel maps to correspond to the new channel that has been added.
212 fChannelMap->insert(std::make_pair(chan->GetAddress(), chan));
213 if((chan->GetNumber() != 0) && (fChannelNumberMap->count(chan->GetNumber()) == 0)) {
214 fChannelNumberMap->insert(std::make_pair(chan->GetNumber(), chan));
215 }
216 }
217}

371 }
372 ++(*fMissingChannelMap)[temp_address];
373 }

504double TChannel::CalibrateENG(int charge, int temp_int) const
505{
506 /// Returns the calibrated energy of the channel when a charge is passed to it.
507 /// This is done by first adding a random number between 0 and 1 to the charge
508 /// bin. This is then taken and divided by the integration parameter. The
509 /// polynomial energy calibration formula is then applied to get the calibrated
510 /// energy.
511 if(charge == 0) {
512 return 0.0000;
513 }
514
515 // We need to add a random number between 0 and 1 before calibrating to avoid
516 // binning issues.
517 return CalibrateENG(static_cast<double>(charge) + gRandom->Uniform(), temp_int);
518}

522 /// Returns the calibrated energy of the channel when a charge is passed to it.
523 /// This is divided by the integration parameter. The
524 /// polynomial energy calibration formula is then applied to get the calibrated

541double TChannel::CalibrateENG(double charge) const
542{
543 /// Returns the calibrated energy. The polynomial energy calibration formula is
544 /// applied to get the calibrated energy. This function does not use the
545 /// integration parameter. If multiple ranges are present, the calibration of the
546 /// range is used. In overlap regions the range used is chosen randomly.
547 if(fENGCoefficients.empty()) {
548 return charge;
549 }
550 // select range to use, they should be sorted
551 size_t currentRange = 0;
552 if(!fENGRanges.empty()) {
553 // we need a boolean to tell us if we found a range as the test
554 // currentRange+1 == fENGRanges.size()
555 // fails if the charge is in the overlap between the second-to-last and last range
556 // and the last range was selected
557 bool foundRange = false;
558 for(currentRange = 0; currentRange+1 < fENGRanges.size(); ++currentRange) {
559 // check if the current range covers this charge
560 if(fENGRanges.Value()[currentRange].first < charge && charge < fENGRanges.Value()[currentRange].second) {
561 // check if there is an overlap with the next range in which case we select that one in 50% of the cases
562 if(fENGRanges.Value()[currentRange+1].first < charge && charge < fENGRanges.Value()[currentRange+1].second) {
563 if(gRandom->Uniform() > 0.5) ++currentRange;
564 }
565 foundRange = true;
566 break;
567 }
568 }
569 if(!foundRange) {
570 currentRange = fENGRanges.size()-1;
571 if(charge < fENGRanges[currentRange].first || fENGRanges[currentRange].second < charge) {
572 std::cerr<<"Charge "<<charge<<" outside all ranges of calibration (first "<<fENGRanges.front().first<<" - "<<fENGRanges.front().second<<", last "<<fENGRanges.back().first<<" - "<<fENGRanges.back().second<<")"<<std::endl;
573 }
574 }
575 }
576
577 // apply the drift correction first
578 if(!fENGDriftCoefficents.empty()) {
579 double corrCharge = -fENGDriftCoefficents[0]; // ILL subtracts the offset instead of adding it
580 for(size_t i = 1; i < fENGDriftCoefficents.size(); i++) {
581 corrCharge += fENGDriftCoefficents[i] * pow(charge, i);
582 }
583 charge = corrCharge;
584 }
585 // if we had drift correction charge is now the corrected charge, otherwise it is still the charge
586 double cal_chg = fENGCoefficients[currentRange][0];
587 for(size_t i = 1; i < fENGCoefficients[currentRange].size(); i++) {
588 cal_chg += fENGCoefficients[currentRange][i] * pow((charge), i);
589 }
590 return cal_chg;
591}

1077 linenumber++;
1078 trimWS(line);
1079 size_t comment = line.find("//");
1080 if(comment != std::string::npos) {

1085 }

1106 }
1107 } else {
1108 delete channel;
1109 }
1110 channel = nullptr;
1111 name.clear();
1112 }

1114 if(openbrace != std::string::npos) {
1115 brace_open = true;

1117 trimWS(name);
1118 channel = new TChannel("");

1120 }
1121 //*************************************//
1122 if(brace_open) {
1123 size_t ntype = line.find(':');

1126 line = line.substr(ntype + 1, line.length());
