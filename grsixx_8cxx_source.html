73static bool StayUp(int milliSec)
74{
75 // Returns false if milliSec milliseconds have passed since logo
76 // was popped up, true otherwise.
77
78 struct timeval ctv, dtv, tv, ptv = gPopupTime;
79
80 tv.tv_sec = milliSec / 1000;
81 tv.tv_usec = (milliSec % 1000) * 1000;
82
83 gettimeofday(&ctv, nullptr);
84 if((dtv.tv_usec = ctv.tv_usec - ptv.tv_usec) < 0) {
85 dtv.tv_usec += 1000000;
86 ptv.tv_sec++;
87 }
88 dtv.tv_sec = ctv.tv_sec - ptv.tv_sec;
89
90 if((ctv.tv_usec = tv.tv_usec - dtv.tv_usec) < 0) {
91 ctv.tv_usec += 1000000;
92 dtv.tv_sec++;
93 }
94 ctv.tv_sec = tv.tv_sec - dtv.tv_sec;
95
96 return ctv.tv_sec >= 0;
97}

112static Pixmap GetRootLogo()
113{
114 // Get logo from xpm file.
115
116 Pixmap logo = 0;
117 Screen* xscreen = XDefaultScreenOfDisplay(gDisplay);
118 if(xscreen == nullptr) {
119 return logo;
120 }
121
122 int depth = PlanesOfScreen(xscreen);
123
124 XWindowAttributes win_attr;
125 XGetWindowAttributes(gDisplay, gLogoWindow, &win_attr);
126
127 XpmAttributes attr;
128 attr.valuemask = XpmVisual | XpmColormap | XpmDepth;
129 attr.visual = win_attr.visual;
130 attr.colormap = win_attr.colormap;
131 attr.depth = win_attr.depth;
132
133#ifdef XpmColorKey // Not available in XPM 3.2 and earlier
134 attr.valuemask |= XpmColorKey;
135 if(depth > 4) {
136 attr.color_key = XPM_COLOR;
137 } else if(depth > 2) {
138 attr.color_key = XPM_GRAY4;
139 } else if(depth > 1) {
140 attr.color_key = XPM_GRAY;
141 } else if(depth == 1) {
142 attr.color_key = XPM_MONO;
143 } else {
144 attr.valuemask &= ~XpmColorKey;
145 }
146
147#endif // defined(XpmColorKey)
148 std::string file;
149 file.append(getenv("GRSISYS"));
150 file.append("/libraries/TGRSIint/grsisplash_bw.xpm");
151
152 int ret = XpmReadFileToPixmap(gDisplay, gLogoWindow, const_cast<char*>(file.c_str()), &logo, nullptr, &attr);
153 XpmFreeAttributes(&attr);
154
155 if(ret == XpmSuccess || ret == XpmColorError) {
156 return logo;
157 }
158
159 std::cout<<"rootx xpm error: "<<XpmGetErrorString(ret)<<std::endl;
160
161 if(logo != 0u) {
162 XFreePixmap(gDisplay, logo);
163 }
164 logo = 0;
165
166 return logo;
167}

169static void ReadContributors()
170{
171 // Read the file $ROOTSYS/README/CREDITS for the names of the
172 // contributors.
173
174 char buf[2048];
175#ifdef ROOTDOCDIR
176 snprintf(buf, sizeof(buf), "%s/CREDITS", ROOTDOCDIR);
177#else
178 snprintf(buf, sizeof(buf), "%s/README/CREDITS", getenv("ROOTSYS"));
179#endif
180
181 gContributors = nullptr;
182
183 FILE* f = fopen(buf, "r");
184 if(f == nullptr) {
185 return;
186 }
187
188 int cnt = 0;
189 while(fgets(buf, sizeof(buf), f) != nullptr) {
190 if(strncmp(buf, "N: ", 3) == 0) {
191 cnt++;
192 }
193 }
194 gContributors = new char*[cnt + 1];
195
196 cnt = 0;
197 rewind(f);
198 while(fgets(buf, sizeof(buf), f) != nullptr) {
199 if(strncmp(buf, "N: ", 3) == 0) {
200 int len = strlen(buf);
201 buf[len - 1] = 0; // remove \n
202 len -= 3; // remove "N: "
203 gContributors[cnt] = new char[len];
204 strncpy(gContributors[cnt], buf + 3, len);
205 cnt++;
206 }
207 }
208 gContributors[cnt] = nullptr;
209
210 fclose(f);
211}

232static int DrawCreditItem(const char* creditItem, const char** members, int y, bool draw)
233{
234 // Draw credit item.
235
236 char credit[1024];
237 int i;
238 int lineSpacing = gFont->max_bounds.ascent + gFont->max_bounds.descent;
239
240 strlcpy(credit, creditItem, sizeof(credit));
241 for(i = 0; (members != nullptr) && (members[i] != nullptr); i++) {
242 if(i != 0) {
243 strlcat(credit, ", ", sizeof(credit));
244 }
245 if(XTextWidth(gFont, credit, strlen(credit)) + XTextWidth(gFont, members[i], strlen(members[i])) >
246 static_cast<int>(gCreditsWidth)) {
247 if(draw) {
248 XDrawString(gDisplay, gCreditsPixmap, gGC, 0, y, credit, strlen(credit));
249 }
250 y += lineSpacing;
251 strlcpy(credit, " ", sizeof(credit));
252 }
253 strlcat(credit, members[i], sizeof(credit));
254 }
255 if(draw) {
256 XDrawString(gDisplay, gCreditsPixmap, gGC, 0, y, credit, strlen(credit));
257 }
258
259 return y;
260}

262static int DrawCredits(bool draw, bool)
263{
264 /// Draw credits. If draw is true draw credits,
265 /// otherwise just return size of all credit text.
266
267 if(gFont == nullptr) {
268 return 150; // size not important no text will be drawn anyway
269 }
270
271 int lineSpacing = gFont->max_bounds.ascent + gFont->max_bounds.descent;
272 int y = lineSpacing;
273
274 y = DrawCreditItem("Conception: ", gConception, y, draw);
275
276 y += 2 * lineSpacing - 1;
277
278 y = DrawCreditItem("Lead Developers: ", gLeadDevelopers, y, draw);
279
280 y += 2 * lineSpacing - 1; // special layout tweak
281
282 y = DrawCreditItem("Key Contributions: ", gKeyContributors, y, draw);
283
284 y += 2 * lineSpacing - 1; // special layout tweak
285
286 y += 10;
287 return y;
288}

302void PopupLogo(bool about)
303{
304 // Popup logo, waiting till ROOT is ready to run.
305 gDisplay = XOpenDisplay("");
306 if(gDisplay == nullptr) {
307 return;
308 }
309
310 gAbout = about;
311
312 Pixel back, fore;
313 int screen = DefaultScreen(gDisplay);
314
315 back = WhitePixel(gDisplay, screen);
316 fore = BlackPixel(gDisplay, screen);
317
318 gLogoWindow = XCreateSimpleWindow(gDisplay, DefaultRootWindow(gDisplay), -100, -100, 50, 50, 0, fore, back);
319
320 gLogoPixmap = GetRootLogo();
321
322 if(gLogoPixmap == 0u) {
323 XCloseDisplay(gDisplay);
324 gDisplay = nullptr;
325 return;
326 }
327
328 Window root;
329 int x, y;
330 unsigned int bw, depth;
331 XGetGeometry(gDisplay, gLogoPixmap, &root, &x, &y, &gWidth, &gHeight, &bw, &depth);
332
333 Screen* xscreen = XDefaultScreenOfDisplay(gDisplay);
334
335 if(xscreen == nullptr) {
336 XCloseDisplay(gDisplay);
337 gDisplay = nullptr;
338 return;
339 }
340 x = (WidthOfScreen(xscreen) - gWidth) / 2;
341 y = (HeightOfScreen(xscreen) - gHeight) / 2;
342
343 XMoveResizeWindow(gDisplay, gLogoWindow, x, y, gWidth, gHeight);
344 XSync(gDisplay, False); // make sure move & resize is done before mapping
345
346 unsigned long valmask;
347 XSetWindowAttributes xswa;
348 valmask = CWBackPixmap | CWOverrideRedirect;
349 xswa.background_pixmap = gLogoPixmap;
350 xswa.override_redirect = True;
351 XChangeWindowAttributes(gDisplay, gLogoWindow, valmask, &xswa);
352
353 gGC = XCreateGC(gDisplay, gLogoWindow, 0, nullptr);
354 gFont = XLoadQueryFont(gDisplay, "-adobe-helvetica-medium-r-*-*-12-*-*-*-*-*-iso8859-1");
355 if(gFont == nullptr) {
356 std::cout<<"Couldn't find font \"-adobe-helvetica-medium-r-*-*-12-*-*-*-*-*-iso8859-1\","<<std::endl
357 <<"trying \"fixed\". Please fix your system so helvetica can be found, "<<std::endl
358 <<"this font typically is in the rpm (or pkg equivalent) package "<<std::endl
359 <<"XFree86-[75,100]dpi-fonts or fonts-xorg-[75,100]dpi."<<std::endl;
360 gFont = XLoadQueryFont(gDisplay, "fixed");
361 if(gFont == nullptr) {
362 std::cout<<"Also couln't find font \"fixed\", your system is terminally misconfigured."<<std::endl;
363 }
364 }
365 if(gFont != nullptr) {
366 XSetFont(gDisplay, gGC, gFont->fid);
367 }
368 XSetForeground(gDisplay, gGC, fore);
369 XSetBackground(gDisplay, gGC, back);
370
371 if(about) {
372 ReadContributors();
373 }
374
375 gCreditsHeight = DrawCredits(false, about) + gCreditsRect.height + 50;
376 gCreditsPixmap = XCreatePixmap(gDisplay, gLogoWindow, gCreditsWidth, gCreditsHeight, depth);
377 XSetForeground(gDisplay, gGC, back);
378 XFillRectangle(gDisplay, gCreditsPixmap, gGC, 0, 0, gCreditsWidth, gCreditsHeight);
379 XSetForeground(gDisplay, gGC, fore);
380 DrawCredits(true, about);
381
382 XSelectInput(gDisplay, gLogoWindow, ButtonPressMask | ExposureMask);
383
384 XMapRaised(gDisplay, gLogoWindow);
385
386 gettimeofday(&gPopupTime, nullptr);
387}

389void WaitLogo()
390{
391 // Main event loop waiting till time arrives to pop down logo
392 // or when forced by button press event.
393
394 if(gDisplay == nullptr) {
395 return;
396 }
397
398 int ypos = 0;
399 bool stopScroll = false;
400
401 ScrollCredits(ypos);
402 DrawVersion();
403 DrawROOTCredit();
404 XFlush(gDisplay);
405
406 while(!gDone) {
407 gSystem->Sleep(3500);
408 break;
409
410 XEvent event;
411 if(XCheckMaskEvent(gDisplay, ButtonPressMask | ExposureMask, &event) != 0) {
412 switch(event.type) {
413 case Expose:
414 if(event.xexpose.count == 0) {
415 ScrollCredits(ypos);
416 DrawVersion();
417 }
418 break;
419 case ButtonPress:
420 if(gAbout && event.xbutton.button == 3) {
421 stopScroll = !stopScroll;
422 } else {
423 gDone = true;
424 }
425 break;
426 default: break;
427 }
428 }
429
430 Sleep(100);
431
432 if(!gAbout && !StayUp(gStayUp) && gMayPopdown) {
433 gDone = true;
434 }
435
436 if(gAbout && !stopScroll) {
437 if(ypos == 0) {
438 Sleep(2000);
439 }
440 ypos++;
441 if(ypos > static_cast<int>(gCreditsHeight - gCreditsRect.height - 50)) {
442 ypos = -int(gCreditsRect.height);
443 }
444 ScrollCredits(ypos);
445 XFlush(gDisplay);
446 }
447 }
448
449 if(gLogoWindow != 0u) {
450 XUnmapWindow(gDisplay, gLogoWindow);
451 XDestroyWindow(gDisplay, gLogoWindow);
452 gLogoWindow = 0;
453 }
454 if(gLogoPixmap != 0u) {
455 XFreePixmap(gDisplay, gLogoPixmap);
456 gLogoPixmap = 0;
457 }
458 if(gCreditsPixmap != 0u) {
459 XFreePixmap(gDisplay, gCreditsPixmap);
460 gCreditsPixmap = 0;
461 }
462 if(gFont != nullptr) {
463 XFreeFont(gDisplay, gFont);
464 gFont = nullptr;
465 }
466 if(gGC != nullptr) {
467 XFreeGC(gDisplay, gGC);
468 gGC = nullptr;
469 }
470 if(gDisplay != nullptr) {
471 XSync(gDisplay, False);
472 XCloseDisplay(gDisplay);
473 gDisplay = nullptr;
474 }
475}
