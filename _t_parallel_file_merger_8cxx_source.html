144Bool_t TParallelFileMerger::Merge()
145{
146 // Merge the current inputs into the output file.
147 DeleteObject(fMerger.GetOutputFile(), false); // Remove object that can *not* be incrementally merge and will *not* be reset by the client code.
148 for(unsigned int f = 0 ; f < fClients.size(); ++f) {
149 fMerger.AddFile(fClients[f].fFile);
150 }
151 Bool_t result = fMerger.PartialMerge(TFileMerger::kAllIncremental);
152
153 // Remove any 'resetable' object (like TTree) from the input file so that they will not
154 // be re-merged. Keep only the object that always need to be re-merged (Histograms).
155 for(unsigned int f = 0 ; f < fClients.size(); ++f) {
156 if(fClients[f].fFile) {
157 DeleteObject(fClients[f].fFile, true);
158 } else {
159 // We back up the file (probably due to memory constraint)
160 TFile* file = TFile::Open(fClients[f].fLocalName, "UPDATE");
161 DeleteObject(file, true); // Remove object that can be incrementally merge and will be reset by the client code.
162 file->Write();
163 delete file;
164 }
165 }
166 fLastMerge = TTimeStamp();
167 fNClientsContact = 0;
168 fClientsContact.Clear();
169
170 return result;
171}

179Bool_t TParallelFileMerger::NeedMerge(Float_t clientThreshold)
180{
181 // Return true, if enough client have reported
182 if(fClients.empty()) {
183 return false;
184 }
185
186 // Calculate average and rms of the time between the last 2 contacts.
187 Double_t sum = 0;
188 Double_t sum2 = 0;
189 for(unsigned int c = 0 ; c < fClients.size(); ++c) {
190 sum += fClients[c].fTimeSincePrevContact;
191 sum2 += fClients[c].fTimeSincePrevContact*fClients[c].fTimeSincePrevContact;
192 }
193 Double_t avg = sum/fClients.size();
194 Double_t sigma = (sum2 != 0) ? TMath::Sqrt(sum2/fClients.size() - avg*avg) : 0;
195 Double_t target = avg + 2*sigma;
196 TTimeStamp now;
197 if((now.AsDouble() - fLastMerge.AsDouble()) > target) {
198 return true;
199 }
200 Float_t cut = clientThreshold * fClients.size();
201 return fClientsContact.CountBits() > cut || fNClientsContact > 2*cut;
202}
