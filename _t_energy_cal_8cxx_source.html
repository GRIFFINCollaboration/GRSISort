44void TEnergyCal::SetNucleus(TNucleus* nuc, Option_t* opt)
45{
46 // Sets the nucleus of the TEnergyCal. This function sets the data points
47 // of the TEnergyCal automatically with the provided nucleus
48 TString optstr = opt;
49 optstr.ToUpper();
50 if((GetNucleus() == nullptr) || optstr.Contains("F")) {
51 TGraphErrors::Clear();
52 TCal::SetNucleus(nuc);
53 for(int i = 0; i < GetNucleus()->NTransitions(); i++) {
54 TGraphErrors::SetPoint(i, 0.0, GetNucleus()->GetTransition(i)->GetEnergy());
55 TGraphErrors::SetPointError(i, 0.0, GetNucleus()->GetTransition(i)->GetEnergyUncertainty());
56 }
57 } else if(GetNucleus() != nullptr) {
58 std::cout<<"Nucleus already exists. Use \"F\" option to overwrite"<<std::endl;
59 }
60
61 SetDefaultTitles();
62 // Sort();
63}
