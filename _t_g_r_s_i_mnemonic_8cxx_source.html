104{
105 std::string name = digitizerName.Value();

107 EDigitizer tmpType = EDigitizer::kDefault;
108 int tmpUnit = 1;

112 } else if(name == "GRF4G") {
113 tmpType = EDigitizer::kGRF4G;

119 tmpType = EDigitizer::kTIG64;
120 tmpUnit = 10;

122 tmpType = EDigitizer::kCaen;
123 tmpUnit = 2;

125 tmpType = EDigitizer::kMadc;
126 tmpUnit = 50;

128 tmpType = EDigitizer::kV1190;
129 tmpUnit = 50;
130 } else if(name == "FMC32") {
131 tmpType = EDigitizer::kFMC32;
132 tmpUnit = 10;
133 } else {

135 }

137 timeStampUnit.Set(tmpUnit, digitizerName.Priority());

140void TGRSIMnemonic::Parse(std::string* name)
141{
142 if((name == nullptr) || name->length() < 9) {
143 if((name->length() < 1) && (name->compare(0, 2, "RF") == 0)) {
144 SetRFMnemonic(name);
145 }
146 return;
147 }
148 TMnemonic::Parse(name);
149 // Enumerating the fSystemString must come last as the details of other parts of
150 // the mnemonic must be known
151 EnumerateSystem();
152
153 if(fSystem == ESystem::kSiLi) {
154 std::string buf;
155 buf.assign(*name, 7, 2);
156 Segment(static_cast<int16_t>(strtol(buf.c_str(), nullptr, 16)));
157 }
158}

205double TGRSIMnemonic::GetTime(Long64_t timestamp, Float_t cfd, double energy, const TChannel* channel) const
206{
207 if(channel == nullptr) {
208 Error("GetTime", "No TChannel provided");
209 return static_cast<double>(timestamp) + gRandom->Uniform();
210 }
211
212 Double_t dTime = 0.;
213 switch(static_cast<EDigitizer>(channel->GetDigitizerType())) {
214 case EDigitizer::kGRF16:
215 case EDigitizer::kFMC32:
216 // we need to zero the lowest 18 bits of the timestamp as those are included in the CFD value
217 // TODO: what happens close to the wrap-around of those 18 bits??? This only happens every 2^18 * 10e-8 so 2.5 ms so 400 Hz
218 dTime = static_cast<Double_t>((timestamp & (~0x3ffff)) * channel->GetTimeStampUnit()) + channel->CalibrateCFD((cfd + gRandom->Uniform()) / 1.6); // CFD is in 10/16th of a nanosecond
219 return dTime - channel->GetTZero(energy) - static_cast<double>(channel->GetTimeOffset());
220 case EDigitizer::kGRF4G:
221 dTime = static_cast<Double_t>(timestamp * channel->GetTimeStampUnit()) + channel->CalibrateCFD((static_cast<Int_t>(cfd) >> 22) + ((static_cast<Int_t>(cfd) & 0x3fffff) + gRandom->Uniform()) / 256.);
222 return dTime - channel->GetTZero(energy) - static_cast<double>(channel->GetTimeOffset());
223 case EDigitizer::kTIG10:
224 dTime = static_cast<Double_t>((timestamp & (~0x7fffff)) * channel->GetTimeStampUnit()) + channel->CalibrateCFD((cfd + gRandom->Uniform()) / 1.6); // CFD is in 10/16th of a nanosecond
225 //channel->CalibrateCFD((cfd & (~0xf) + gRandom->Uniform()) / 1.6); // PBender suggests this.
226 return dTime - channel->GetTZero(energy) - static_cast<double>(channel->GetTimeOffset());
227 case EDigitizer::kCaen:
228 //10 bit CFD for 0-2ns => divide by 512
229 dTime = static_cast<Double_t>(timestamp * channel->GetTimeStampUnit()) + channel->CalibrateCFD((cfd + gRandom->Uniform()) / 512.);
230 return dTime - channel->GetTZero(energy) - static_cast<double>(channel->GetTimeOffset());
231 default:
232 dTime = (static_cast<Double_t>(timestamp) + gRandom->Uniform()) * channel->GetTimeStampUnit();
233 return dTime - channel->GetTZero(energy) - static_cast<double>(channel->GetTimeOffset());
234 }
235 return 0.;
236}

238int TGRSIMnemonic::NumericArraySubPosition() const
239{
240 /// This function translates the crystal color to an index
241 /// B - Blue - 0, G - Green - 1, R - Red - 2, W - White - 3, default - 5
242 /// Except for the LaBr BGOs which use A - 0, B - 1, C - 2
243 /// and a default of 5
244 if(System() == ESystem::kLaBrBgo) {
245 switch(ArraySubPosition()) {
246 case TMnemonic::EMnemonic::kA:
247 return 0;
248 case TMnemonic::EMnemonic::kB:
249 return 1;
250 case TMnemonic::EMnemonic::kC:
251 return 2;
252 default:
253 return 5;
254 };
255 }
256
257 switch(ArraySubPosition()) {
258 case TMnemonic::EMnemonic::kB:
259 return 0;
260 case TMnemonic::EMnemonic::kG:
261 return 1;
262 case TMnemonic::EMnemonic::kR:
263 return 2;
264 case TMnemonic::EMnemonic::kW:
265 return 3;
266 default:
267 return 5;
268 };
269
270 // return statement here instead of default case
271 // to make sure compiler doesn't warn us about missing return
272 return 5;
273}
