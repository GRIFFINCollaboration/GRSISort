238template <class T>
239NRMat<T>& NRMat<T>::operator=(const NRMat<T>& rhs)
240// postcondition: normal assignment via copying has been performed;
241// if matrix and rhs were different sizes, matrix
242// has been resized to match the size of rhs
243{
244 if(this != &rhs) {
245 int i, j;
246 if(nn != rhs.nn || mm != rhs.mm) {
247 if(v != 0) {
248 delete[](v[0]);
249 delete[](v);
250 }
251 nn = rhs.nn;
252 mm = rhs.mm;
253 v = new T*[nn];
254 v[0] = new T[mm * nn];
255 }
256 for(i = 1; i < nn; i++) {
257 v[i] = v[i - 1] + mm;
258 }
259 for(i = 0; i < nn; i++) {
260 for(j = 0; j < mm; j++) {
261 v[i][j] = rhs[i][j];
262 }
263 }
264 }
265 return *this;
266}

336template <class T>
337NRMat3d<T>::NRMat3d(int n, int m, int k) : nn(n), mm(m), kk(k), v(new T**[n])
338{
339 int i, j;
340 v[0] = new T*[n * m];
341 v[0][0] = new T[n * m * k];
342 for(j = 1; j < m; j++) {
343 v[0][j] = v[0][j - 1] + k;
344 }
345 for(i = 1; i < n; i++) {
346 v[i] = v[i - 1] + m;
347 v[i][0] = v[i - 1][0] + m * k;
348 for(j = 1; j < m; j++) {
349 v[i][j] = v[i][j - 1] + k;
350 }
351 }
352}
