28double GetYError(TGraphErrors* graph, const double& x)
29{
30 /// general function to get the error of a graph at point x (takes the maximum of the errors of the bracketing points)
31 for(int i = 0; i < graph->GetN(); ++i) {
32#if ROOT_VERSION_CODE >= ROOT_VERSION(6,20,0)
33 // exact match: return the error at this point
34 if(graph->GetPointX(i) == x) return graph->GetErrorY(i);
35 // first point with larger x: take maximum of this point and previous point
36 // TGraphErrors::GetErrorY returns -1 if index is negative, so we don't need to check for this
37 else if(graph->GetPointX(i) > x) return std::max(graph->GetErrorY(i-1), graph->GetErrorY(i));
38#else
39 double px, py;
40 graph->GetPoint(i, px, py);
41 // exact match: return the error at this point
42 if(px == x) return graph->GetErrorY(i);
43 // first point with larger x: take maximum of this point and previous point
44 // TGraphErrors::GetErrorY returns -1 if index is negative, so we don't need to check for this
45 else if(px > x) return std::max(graph->GetErrorY(i-1), graph->GetErrorY(i));
46#endif
47 }
48 return 0.;
49}

648class Ac {
649public:
650 Ac(TGraphErrors* data = nullptr, TGraphErrors* z0 = nullptr, TGraphErrors* z2 = nullptr, TGraphErrors* z4 = nullptr)
651 : fData(data), fZ0(z0), fZ2(z2), fZ4(z4) {}
652
653 void Data(TGraphErrors* data) { fData = data; }
654 void Z0(TGraphErrors* z0) { fZ0 = z0; }
655 void Z2(TGraphErrors* z2) { fZ2 = z2; }
656 void Z4(TGraphErrors* z4) { fZ4 = z4; }
657 void SetZ(TGraphErrors* z0, TGraphErrors* z2, TGraphErrors* z4) { fZ0 = z0; fZ2 = z2; fZ4 = z4; }
658
659 int Np() { return fData->GetN(); }
660
661 double operator()(const double* p) {
662 double chi2 = 0;
663 for(int point = 0; point < fData->GetN(); ++point) {
664 // get data values
665 double x;
666 double y;
667 fData->GetPoint(point, x, y);
668 double yError = fData->GetErrorY(point);
669
670 // get simulation values
671 // for the y-values we can use Eval (uses linear interpolation)
672 double functionValue = p[0]*((1.-p[1]-p[2])*fZ0->Eval(x) + p[1]*fZ2->Eval(x) + p[2]*fZ4->Eval(x));
673 // for the y uncertainties we need to find the index (for each graph)
674 double errorSquare = TMath::Power(yError, 2) + TMath::Power(p[0], 2)*(TMath::Power((1.-p[1]-p[2])*GetYError(fZ0, x), 2) + TMath::Power(p[1]*GetYError(fZ2, x), 2) + TMath::Power(p[2]*GetYError(fZ4, x), 2));
675
676 // calculate chi^2
677 chi2 += TMath::Power((y-functionValue), 2)/errorSquare;
678 }
679 return chi2;
680 }
681
682private:
683 TGraphErrors* fData{nullptr};
684 TGraphErrors* fZ0{nullptr};
685 TGraphErrors* fZ2{nullptr};
686 TGraphErrors* fZ4{nullptr};
687};

689TGraph* MixingMethod(TGraphErrors* data, TGraphErrors* z0, TGraphErrors* z2, TGraphErrors* z4, int twoJhigh, int twoJmid, int twoJlow, std::vector<double>& bestParameters)
690{
691 TGraph* result = nullptr;
692 Ac ac(data, z0, z2, z4);
693 ROOT::Fit::Fitter fitter;
694 int nPar = 3;
695 fitter.SetFCN(nPar, ac);
696 for(int i = 0; i < nPar; ++i) {
697 // parameter settings arguments are parameter name, initial value, step size, minimum, and maximum
698 fitter.Config().ParSettings(i) = ROOT::Fit::ParameterSettings(Form("a_{%d}", 2*i), 0.5, 0.0001, -10., 10.);
699 }
700 fitter.Config().MinimizerOptions().SetPrintLevel(0);
701 fitter.Config().SetMinimizer("Minuit2", "Migrad"); // or simplex?
702
703 // j1 is the spin of the highest level
704 // j2 is the spin of the middle level
705 // j3 is the spin of the bottom level
706 double j1 = 0.5*twoJhigh;
707 double j2 = 0.5*twoJmid;
708 double j3 = 0.5*twoJlow;
709 // l1 is the transition between j1 and j2
710 // a is the lowest allowed spin
711 // b is the mixing spin
712 int l1a = TMath::Abs(twoJhigh-twoJmid)/2;
713 if(l1a == 0) l1a = 1;
714 int l1b = l1a + 1;
715 // l2 is the transition between j2 and j3
716 // a is the lowest allowed spin
717 // b is the mixing spin
718 int l2a = TMath::Abs(twoJmid-twoJlow)/2;
719 if(l2a == 0) l2a = 1;
720 int l2b = l2a + 1;
721
722 // run some quick checks on the mixing ratios
723 if((twoJhigh==0 && twoJmid==0) || (twoJmid==0 && twoJlow==0)) {
724 std::cout<<"!!!!!!!!!!!!!!! ERROR !!!!!!!!!!!!!!!"<<std::endl
725 <<"Can't have gamma transition between J=0 states (high "<<twoJhigh<<", mid "<<twoJmid<<", low "<<twoJlow<<")."<<std::endl
726 <<"Aborting..."<<std::endl;
727 return result;
728 }
729 if(l1a==TMath::Abs(twoJhigh+twoJmid)/2) {
730 //std::cout<<"!!!!!!!!!!!!!!! ALERT !!!!!!!!!!!!!!!"<<std::endl
731 // <<"Only one angular momentum allowed for high->middle transition (l1a "<<l1a<<" == "<<TMath::Abs(twoJhigh+twoJmid)/2<<")."<<std::endl
732 // <<"That mixing ratio (delta1) will be fixed at zero."<<std::endl
733 // <<"!!!!!!!!!!!!! END ALERT !!!!!!!!!!!!!"<<std::endl;
734 l1b = l1a;
735 }
736 if(l2a==TMath::Abs(twoJmid+twoJlow)/2) {
737 //std::cout<<"!!!!!!!!!!!!!!! ALERT !!!!!!!!!!!!!!!"<<std::endl
738 // <<"Only one angular momentum allowed for middle->low transition (l2a "<<l2a<<" == "<<TMath::Abs(twoJmid+twoJlow)/2<<")."<<std::endl
739 // <<"That mixing ratio (delta2) will be fixed at zero."<<std::endl
740 // <<"!!!!!!!!!!!!! END ALERT !!!!!!!!!!!!!"<<std::endl;
741 l2b = l2a;
742 }
743
744 // -------------------------------------------------------------------//
745 // Constrained fitting
746 // -------------------------------------------------------------------//
747 // The basic idea is to select a particular set of physical quantities,
748 // calculate a2/a4, fix the a2/a4 parameters, and fit the scaling
749 // factor a0. Then output the specifications for that set of physical
750 // quantities and the chi^2 for further analysis.
751 // -------------------------------------------------------------------//
752
753 // delta runs from -infinity to infinity (unless constrained by known physics)
754 // in this case, it then makes more sense to sample evenly from tan^{-1}(delta)
755 // The next few lines are where you may want to include limits to significantly speed up calculations
756 // mixing for the high-middle transition
757 double mixingAngle1Minimum = -TMath::Pi()/2;
758 double mixingAngle1Maximum = TMath::Pi()/2;
759 int steps1 = 100;
760 double stepSize1 = (mixingAngle1Maximum-mixingAngle1Minimum)/steps1;
761 // mixing for the middle-low transition
762 double mixingAngle2Minimum = -TMath::Pi()/2;
763 double mixingAngle2Maximum = TMath::Pi()/2;
764 int steps2 = 100;
765 double stepSize2 = (mixingAngle2Maximum-mixingAngle2Minimum)/steps2;
766
767 // if appropriate, constrain the delta values
768 if(l1a==l1b) { mixingAngle1Minimum = 0; steps1 = 1; }
769 if(l2a==l2b) { mixingAngle2Minimum = 0; steps2 = 1; }
770
771 result = new TGraph(steps1);
772 double minChi2 = 1e6;
773 for(int i=0; i < steps1; i++) {
774 double mixangle1 = mixingAngle1Minimum + i*stepSize1;
775 double delta1 = TMath::Tan(mixangle1);
776 for(int j=0; j < steps2; j++) {
777 double mixangle2 = mixingAngle2Minimum + j*stepSize2;
778 double delta2 = TMath::Tan(mixangle2);
779 // calculate a2
780 double a2 = TGRSIFunctions::CalculateA2(j1,j2,j3,l1a,l1b,l2a,l2b,delta1,delta2);
781 // fix a2
782 fitter.Config().ParSettings(1).Set("a_{2}", a2);
783 fitter.Config().ParSettings(1).Fix();
784 // calculate a4
785 double a4 = TGRSIFunctions::CalculateA4(j1,j2,j3,l1a,l1b,l2a,l2b,delta1,delta2);
786 // fix a4
787 fitter.Config().ParSettings(2).Set("a_{4}", a4);
788 fitter.Config().ParSettings(2).Fix();
789 if(!fitter.FitFCN()) {
790 std::cerr<<i<<", "<<j<<": Fit failed using a2 "<<a2<<", a4 "<<a4<<std::endl;
791 continue;
792 }
793 auto fitResult = fitter.Result();
794 // MinFcnValue() is the minimum chi2, ac.Np gives the number of data points
795 result->SetPoint(i, mixangle1, fitResult.MinFcnValue()/(ac.Np()-fitResult.NFreeParameters()));
796 if(fitResult.MinFcnValue()/(ac.Np()-fitResult.NFreeParameters()) < minChi2) {
797 minChi2 = fitResult.MinFcnValue()/(ac.Np()-fitResult.NFreeParameters());
798 bestParameters.assign(fitResult.GetParams(), fitResult.GetParams()+nPar);
799 }
800 }
801 }
802 return result;
803}
