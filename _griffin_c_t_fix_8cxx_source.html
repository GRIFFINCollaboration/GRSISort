16double CrossTalkFit(double* x, double* par)
17{
18 // This function is the linear fit function, but uses the CT coefficients as parameters instead of slope and
19 // intercept
20 double k0 = par[0] / (1. - par[0]);
21 double k1 = par[1] / (1. - par[1]);
22 double energy = par[2];
23
24 double slope = -(1. + k0) / (1. + k1);
25 double intercept = energy * (1. + k0 / (1. + k1));
26
27 return x[0] * slope + intercept;
28}
