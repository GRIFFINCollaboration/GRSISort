111void TestHelper::Exec(unsigned int slot, TGriffin& grif, TGriffinBgo& grifBgo, TZeroDegree& zds, TSceptar& scep) // NOLINT
112{
113 // we use .at() here instead of [] so that we get meaningful error message if a histogram we try to fill wasn't
114 // created e.g. because of a typo
115
116 // multiplicities
117 fH2[slot].at("zdsMultGriffinMult")->Fill(grif.GetMultiplicity(), zds.GetMultiplicity());
118
119 // loop over unsuppressed griffin hits
120 for(int g = 0; g < grif.GetMultiplicity(); ++g) {
121 auto grif1 = grif.GetGriffinHit(g);
122 fH1[slot].at("griffinE")->Fill(grif1->GetEnergy());
123 }
124
125 // loop over suppressed griffin hits
126 for(int g = 0; g < grif.GetSuppressedMultiplicity(&grifBgo); ++g) {
127 auto grif1 = grif.GetSuppressedHit(g);
128 fH1[slot].at("griffinESupp")->Fill(grif1->GetEnergy());
129 }
130
131 // loop over unsuppressed griffin addback hits
132 for(int g = 0; g < grif.GetAddbackMultiplicity(); ++g) {
133 auto grif1 = grif.GetAddbackHit(g);
134 fH1[slot].at("griffinEAddback")->Fill(grif1->GetEnergy());
135 }
136
137 // loop over suppressed griffin addback hits
138 for(int g = 0; g < grif.GetSuppressedAddbackMultiplicity(&grifBgo); ++g) {
139 auto grif1 = grif.GetSuppressedAddbackHit(g);
140 fH1[slot].at("griffinESuppAddback")->Fill(grif1->GetEnergy());
141 // we use a flag to check if there is any beta in coincidence with this gamma-ray and only fill afterwards
142 // otherwise we could double-count gamma-rays that are in coincidence with more than one beta
143 // e.g. because of double triggering of SCEPTAR/ZDS where we get a second trigger from the tail of the real signal
144 bool promptBeta = false;
145 for(int z = 0; z < zds.GetMultiplicity(); ++z) {
146 auto zds1 = zds.GetZeroDegreeHit(z);
147 fH2[slot].at("griffinZdsTS")->Fill(grif1->GetTimeStampNs() - zds1->GetTimeStampNs(), grif1->GetArrayNumber());
148 fH2[slot].at("griffinZdsTime")->Fill(grif1->GetTime() - zds1->GetTime(), grif1->GetArrayNumber());
149 if(PromptCoincidence(grif1, zds1)) promptBeta = true;
150 }
151 if(promptBeta) {
152 fH1[slot].at("griffinESuppAddbackBeta")->Fill(grif1->GetEnergy());
153 if(fCycleLength > 0.) {
154 fH2[slot].at("griffinCycle")->Fill(std::fmod(grif1->GetTime(), fCycleLength), grif1->GetEnergy());
155 }
156 for(int g2 = g + 1; g2 < grif.GetSuppressedAddbackMultiplicity(&grifBgo); ++g2) {
157 auto grif2 = grif.GetSuppressedAddbackHit(g2);
158 fH2[slot]
159 .at("griffinGriffinTS")
160 ->Fill(grif1->GetTimeStampNs() - grif2->GetTimeStampNs(), grif1->GetArrayNumber());
161 fH2[slot].at("griffinGriffinTime")->Fill(grif1->GetTime() - grif2->GetTime(), grif1->GetArrayNumber());
162 if(PromptCoincidence(grif1, grif2)) {
163 // fill twice to get a symmetric matrix
164 fH2[slot].at("griffinESuppAddbackMatrixBeta")->Fill(grif1->GetEnergy(), grif2->GetEnergy());
165 fH2[slot].at("griffinESuppAddbackMatrixBeta")->Fill(grif2->GetEnergy(), grif1->GetEnergy());
166 } else if(TimeRandom(grif1, grif2)) {
167 // fill twice to get a symmetric matrix
168 fH2[slot].at("griffinESuppAddbackMatrixBetaBg")->Fill(grif1->GetEnergy(), grif2->GetEnergy());
169 fH2[slot].at("griffinESuppAddbackMatrixBetaBg")->Fill(grif2->GetEnergy(), grif1->GetEnergy());
170 }
171 }
172 }
173 }
174 if(fCycleLength > 0.) {
175 for(int z = 0; z < zds.GetMultiplicity(); ++z) {
176 auto zds1 = zds.GetZeroDegreeHit(z);
177 fH1[slot].at("zdsCycle")->Fill(std::fmod(zds1->GetTime(), fCycleLength));
178 }
179 }
