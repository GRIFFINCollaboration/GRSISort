107Bool_t TCalManager::AddToManager(TCal* cal, UInt_t chanNum, Option_t* opt)
108{
109 /// Makes a Deep copy of cal and adds it to the CalManager Map for channel number
110 /// chanNum.
111 if(cal == nullptr) {
112 return false;
113 }
114
115 if(fClass == nullptr) {
116 SetClass(cal->ClassName());
117 } else if(fClass->GetName() != cal->ClassName()) {
118 Error("AddToManager", "Trying to put a %s in a TCalManager of type %s", cal->ClassName(), fClass->GetName());
119 return false;
120 }
121
122 if((cal->GetChannel()) == nullptr) {
123 if(!(cal->SetChannel(chanNum))) {
124 return false; // TCal does the Error for us.
125 }
126 }
127
128 if(fCalMap.count(cal->GetChannel()->GetNumber()) == 1) { // if this cal already exists
129 if(strcmp(opt, "overwrite") == 0) {
130 TCal* oldCal = GetCal(chanNum);
131 // delete the old calibration for this channel number
132 delete oldCal;
133 oldCal = static_cast<TCal*>(cal->Clone(cal->GetName()));
134 fCalMap.at(chanNum) = oldCal;
135 return true;
136 }
137 Error("AddToManager", "Trying to add a channel that already exists!");
138 return false;
139 }
140 TCal* newCal = static_cast<TCal*>(cal->Clone(cal->GetName()));
141 // In order to construct a new derived class you need to know the type at compile time.
142 // Clone lets us get around this. There are other ways to do this using "virtual constructor idioms"
143 // but the basically do what clone does anyway.
144 // Clone uses ROOT streamers. We have made the TChannel part of the TCal a TRef.
145 // This has the effect of making it persistent as far as the ROOT streamer
146 // facility is concerned. All of the other "pointer members" of the TCal
147 // Get Deep copied into the TCal Manager.
148 std::cout << "newCal: " << newCal->GetChannel() << ", cal: " << cal->GetChannel() << std::endl;
149 ;
150 fCalMap.insert(std::make_pair(chanNum, newCal));
151
152 return true;
153}
