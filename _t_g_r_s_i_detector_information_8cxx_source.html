102void TGRSIDetectorInformation::Set()
103{
104 /// Sets the run info. This figures out what systems are available.
105 if(TChannel::GetChannelMap() == nullptr) {
106 std::cerr << RED << "TGRSIDetectorInformation::Set(): Failed to get channel map!" << RESET_COLOR << std::endl;
107 return;
108 }
109 for(auto& iter : *(TChannel::GetChannelMap())) {
110 std::string channelname = iter.second->GetName();
111
112 // check if we have an old TIG digitizer, in that case sort by trigger ID (instead of time stamp)
113 // don't need to check again if we already found one, so include check for fSortByTriggerId being true
114 if(!fSortByTriggerId && (iter.second->GetDigitizerType() == EDigitizer::kTIG10 || iter.second->GetDigitizerType() == EDigitizer::kTIG64)) {
115 fSortByTriggerId = true;
116 }
117 // detector system type.
118 // for more info, see: https://www.triumf.info/wiki/tigwiki/index.php/Detector_Nomenclature
119 switch(static_cast<const TGRSIMnemonic*>(iter.second->GetMnemonic())->System()) {
120 case TGRSIMnemonic::ESystem::kTigress:
121 SetTigress();
122 break;
123 case TGRSIMnemonic::ESystem::kSharc:
124 SetSharc();
125 break;
126 case TGRSIMnemonic::ESystem::kTriFoil:
127 SetTriFoil();
128 break;
129 case TGRSIMnemonic::ESystem::kRF:
130 SetRF();
131 break;
132 case TGRSIMnemonic::ESystem::kCSM:
133 SetCSM();
134 break;
135 case TGRSIMnemonic::ESystem::kTip:
136 SetTip();
137 break;
138 case TGRSIMnemonic::ESystem::kGriffin:
139 SetGriffin();
140 break;
141 case TGRSIMnemonic::ESystem::kSceptar:
142 SetSceptar();
143 break;
144 case TGRSIMnemonic::ESystem::kPaces:
145 SetPaces();
146 break;
147 case TGRSIMnemonic::ESystem::kLaBr:
148 SetDante();
149 break;
150 case TGRSIMnemonic::ESystem::kZeroDegree:
151 SetZeroDegree();
152 break;
153 case TGRSIMnemonic::ESystem::kDescant:
154 SetDescant();
155 break;
156 case TGRSIMnemonic::ESystem::kBgo:
157 SetBgo();
158 break;
159 case TGRSIMnemonic::ESystem::kEmma:
160 SetEmma();
161 break;
162 case TGRSIMnemonic::ESystem::kTrific:
163 SetTrific();
164 break;
165 case TGRSIMnemonic::ESystem::kSharc2:
166 SetSharc2();

168 case TGRSIMnemonic::ESystem::kRcmp:

180 default:
181 std::string system = iter.second->GetMnemonic()->SystemString();
182 if(system == "SP") {
183 SetSpice();
184 SetS3();

186 SetBambino();
187 }
188 };
