95Double_t TABPeak::OneHitPeakFunction(Double_t *dim, Double_t *par)
96{
97 Double_t x = dim[0]; // channel number used for fitting
98 Double_t height = par[0]; // height of photopeak
99 Double_t c = par[1]; // Peak Centroid of non skew gaus
100 Double_t sigma = par[2]; // standard deviation of gaussian
101
102 return height * TMath::Gaus(x, c, sigma);
103}

105Double_t TABPeak::TwoHitPeakFunction(Double_t *dim, Double_t *par)
106{
107 Double_t x = dim[0]; // channel number used for fitting
108 Double_t height = par[0]; // height of photopeak
109 Double_t c = par[1]; // Peak Centroid of non skew gaus
110 Double_t sigma = par[2]; // standard deviation of gaussian
111 Double_t rel_height = par[3]; // relative height of 2-hit peak
112 Double_t rel_sigma = par[4]; // relative sigma of 2-hit peak
113
114 return height * rel_height * TMath::Gaus(x,c,rel_sigma*sigma);
115}

127Double_t TABPeak::BackgroundFunction(Double_t *dim, Double_t *par)
128{
129 Double_t x = dim[0]; // channel number used for fitting
130 Double_t height = par[0]; // height of photopeak
131 Double_t c = par[1]; // Peak Centroid of non skew gaus
132 Double_t sigma = par[2]; // standard deviation of gaussian
133 Double_t step = par[5]; // Size of the step function;
134
135 Double_t step_func = TMath::Abs(step) * height / 100.0 * TMath::Erfc((x - c) / (TMath::Sqrt(2.0) * sigma));
136
137 return step_func;
138}
