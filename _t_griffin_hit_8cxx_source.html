99void TGriffinHit::Add(const TDetectorHit* hit)
100{
101 const TGriffinHit* griffinHit = dynamic_cast<const TGriffinHit*>(hit);
102 if(griffinHit == nullptr) {
103 throw std::runtime_error("trying to add non-griffin hit to griffin hit!");
104 }
105 // add another griffin hit to this one (for addback),
106 // using the time and position information of the one with the higher energy
107 if(!CompareEnergy(this, griffinHit)) {
108 SetCfd(griffinHit->GetCfd());
109 SetTime(griffinHit->GetTime());
110 // SetPosition(griffinHit->GetPosition());
111 SetAddress(griffinHit->GetAddress());
112 } else {
113 SetTime(GetTime());
114 }
115 SetEnergy(GetEnergy() + griffinHit->GetEnergy());
116 // this has to be done at the very end, otherwise GetEnergy() might not work
117 SetCharge(0);
118 // Add all of the pileups.This should be changed when the max number of pileups changes
119 if((NPileUps() + griffinHit->NPileUps()) < 4) {
120 SetNPileUps(NPileUps() + griffinHit->NPileUps());
121 } else {
122 SetNPileUps(3);
123 }
124 if((PUHit() + griffinHit->PUHit()) < 4) {
125 SetPUHit(PUHit() + griffinHit->PUHit());
126 } else {
127 SetPUHit(3);
128 }
129 // KValue is somewhate meaningless in addback, so I am using it as an indicator that a piledup griffinHit was added-back RD
130 if(GetKValue() > griffinHit->GetKValue()) {
131 SetKValue(griffinHit->GetKValue());
132 }
133}
