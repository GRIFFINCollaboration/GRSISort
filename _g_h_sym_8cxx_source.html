75Int_t GHSym::BufferEmpty(Int_t action)
76{
77 /// Fill histogram with all entries in the buffer.
78 /// action = -1 histogram is reset and refilled from the buffer (called by THistPainter::Paint)
79 /// action = 0 histogram is filled from the buffer
80 /// action = 1 histogram is filled and buffer is deleted
81 /// The buffer is automatically deleted when the number of entries
82 /// in the buffer is greater than the number of entries in the histogram
83 if(fBuffer == nullptr) {
84 return 0;
85 }
86
87 auto nbEntries = static_cast<Int_t>(fBuffer[0]);
88 if(nbEntries == 0) {
89 return 0;
90 }
91 if(nbEntries < 0 && action == 0) {
92 return 0; // histogram has been already filled from the buffer
93 }
94 Double_t* buffer = fBuffer;
95 if(nbEntries < 0) {
96 nbEntries = -nbEntries;
97 fBuffer = nullptr;
98 Reset("ICES");
99 fBuffer = buffer;
100 }
101
102 if(CanExtendAllAxes() || fXaxis.GetXmax() <= fXaxis.GetXmin() || fYaxis.GetXmax() <= fYaxis.GetXmin()) {
103 // find min, max of entries in buffer
104 // for the symmetric matrix x- and y-range are the same
105 Double_t min = fBuffer[2];
106 Double_t max = min;
107 if(fBuffer[3] < min) {
108 min = fBuffer[3];
109 }
110 if(fBuffer[3] > max) {
111 max = fBuffer[3];
112 }
113 for(Int_t i = 1; i < nbEntries; ++i) {
114 Double_t x = fBuffer[3 * i + 2];
115 if(x < min) {
116 min = x;
117 }
118 if(x > max) {
119 max = x;
120 }
121 Double_t y = fBuffer[3 * i + 3];
122 if(y < min) {
123 min = y;
124 }
125 if(y > max) {
126 max = y;
127 }
128 }
129 if(fXaxis.GetXmax() <= fXaxis.GetXmin() || fYaxis.GetXmax() <= fYaxis.GetXmin()) {
130 THLimitsFinder::GetLimitsFinder()->FindGoodLimits(this, min, max, min, max);
131 } else {
132 fBuffer = nullptr;
133 Int_t keep = fBufferSize;
134 fBufferSize = 0;
135 if(min < fXaxis.GetXmin()) {
136 RebinAxis(min, &fXaxis);
137 }
138 if(max >= fXaxis.GetXmax()) {
139 RebinAxis(max, &fXaxis);
140 }
141 if(min < fYaxis.GetXmin()) {
142 RebinAxis(min, &fYaxis);
143 }
144 if(max >= fYaxis.GetXmax()) {
145 RebinAxis(max, &fYaxis);
146 }
147 fBuffer = buffer;
148 fBufferSize = keep;
149 }
150 }
151
152 fBuffer = nullptr;
153 for(Int_t i = 0; i < nbEntries; ++i) {
154 Fill(buffer[3 * i + 2], buffer[3 * i + 3], buffer[3 * i + 1]);
155 }
156 fBuffer = buffer;
157
158 if(action > 0) {
159 delete[] fBuffer;
160 fBuffer = nullptr;
161 fBufferSize = 0;
162 } else {
163 if(nbEntries == static_cast<Int_t>(fEntries)) {
164 fBuffer[0] = -nbEntries;
165 } else {
166 fBuffer[0] = 0;
167 }
168 }
169 return nbEntries;
170}

212Double_t GHSym::DoIntegral(Int_t binx1, Int_t binx2, Int_t biny1, Int_t biny2, Double_t& error, Option_t* option, Bool_t doError) const
213{
214 // internal function compute integral and optionally the error between the limits
215 // specified by the bin number values working for all histograms (1D, 2D and 3D)
216
217 Int_t nbinsx = GetNbinsX();
218 if(binx1 < 0) {
219 binx1 = 0;
220 }
221 if(binx2 > nbinsx + 1 || binx2 < binx1) {
222 binx2 = nbinsx + 1;
223 }
224 if(GetDimension() > 1) {
225 Int_t nbinsy = GetNbinsY();
226 if(biny1 < 0) {
227 biny1 = 0;
228 }
229 if(biny2 > nbinsy + 1 || biny2 < biny1) {
230 biny2 = nbinsy + 1;
231 }
232 } else {
233 biny1 = 0;
234 biny2 = 0;
235 }
236
237 // - Loop on bins in specified range
238 TString opt = option;
239 opt.ToLower();
240 Bool_t width = kFALSE;
241 if(opt.Contains("width")) {
242 width = kTRUE;
243 }
244
245 Double_t dx = 1.;
246 Double_t dy = 1.;
247 Double_t integral = 0;
248 Double_t igerr2 = 0;
249 for(Int_t binx = binx1; binx <= binx2; ++binx) {
250 if(width) {
251 dx = fXaxis.GetBinWidth(binx);
252 }
253 for(Int_t biny = biny1; biny <= biny2; ++biny) {
254 if(width) {
255 dy = fYaxis.GetBinWidth(biny);
256 }
257 Int_t bin = GetBin(binx, biny);
258 if(width) {
259 integral += GetBinContent(bin) * dx * dy;
260 } else {
261 integral += GetBinContent(bin);
262 }
263 if(doError) {
264 if(width) {
265 igerr2 += GetBinError(bin) * GetBinError(bin) * dx * dx * dy * dy;
266 } else {
267 igerr2 += GetBinError(bin) * GetBinError(bin);
268 }
269 }
270 }
271 }
272
273 if(doError) {
274 error = TMath::Sqrt(igerr2);
275 }
276 return integral;
277}

380Int_t GHSym::Fill(const char* namex, const char* namey, Double_t w)
381{
382 // Increment cell defined by namex,namey by a weight w
383 //
384 // if x or/and y is less than the low-edge of the corresponding axis first bin,
385 // the Underflow cell is incremented.
386 // if x or/and y is greater than the upper edge of corresponding axis last bin,
387 // the Overflow cell is incremented.
388 //
389 // If the storage of the sum of squares of weights has been triggered,
390 // via the function Sumw2, then the sum of the squares of weights is incremented
391 // by w^2 in the cell corresponding to x,y.
392 //
393
394 Int_t binx = 0;
395 Int_t biny = 0;
396 fEntries++;
397 binx = fXaxis.FindBin(namex);
398 biny = fYaxis.FindBin(namey);
399 if(binx < 0 || biny < 0) {
400 return -1;
401 }
402 if(biny >= binx) {
403 std::swap(binx, biny);
404 }
405 Int_t bin = biny * (2 * fXaxis.GetNbins() - biny + 3) / 2 + binx;
406 AddBinContent(bin, w);
407 if(fSumw2.fN != 0) {
408 fSumw2.fArray[bin] += w * w;
409 }
410 if(binx == 0 || binx > fXaxis.GetNbins()) {
411 return -1;
412 }
413 if(biny == 0 || biny > fYaxis.GetNbins()) {
414 return -1;
415 }
416 Double_t x = fXaxis.GetBinCenter(binx);
417 Double_t y = fYaxis.GetBinCenter(biny);
418 fTsumw += w;
419 fTsumw2 += w * w;
420 fTsumwx += w * x;
421 fTsumwx2 += w * x * x;
422 fTsumwy += w * y;
423 fTsumwy2 += w * y * y;
424 fTsumwxy += w * x * y;
425 return bin;
426}

428void GHSym::FillN(Int_t ntimes, const Double_t* x, const Double_t* y, const Double_t* w, Int_t stride)
429{
430 //*-*-*-*-*-*-*Fill a 2-D histogram with an array of values and weights*-*-*-*
431 //*-* ========================================================
432 //*-*
433 //*-* ntimes: number of entries in arrays x and w (array size must be ntimes*stride)
434 //*-* x: array of x values to be histogrammed
435 //*-* y: array of y values to be histogrammed
436 //*-* w: array of weights
437 //*-* stride: step size through arrays x, y and w
438 //*-*
439 //*-* If the storage of the sum of squares of weights has been triggered,
440 //*-* via the function Sumw2, then the sum of the squares of weights is incremented
441 //*-* by w[i]^2 in the cell corresponding to x[i],y[i].
442 //*-* if w is NULL each entry is assumed a weight=1
443 //*-*
444 //*-* NB: function only valid for a TH2x object
445 //*-*
446 //*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*
447 ntimes *= stride;
448 for(int i = 0; i < ntimes; i += stride) {
449 if(w != nullptr) {
450 Fill(x[i], y[i], w[i]);
451 } else {
452 Fill(x[i], y[i]);
453 }
454 }

457void GHSym::FillRandom(const char* fname, Int_t ntimes, TRandom* rng)
458{
459 //*-*-*-*-*-*-*Fill histogram following distribution in function fname*-*-*-*
460 //*-* =======================================================
461 //*-*
462 //*-* The distribution contained in the function fname (TF2) is integrated
463 //*-* over the channel contents.
464 //*-* It is normalized to 1.
465 //*-* Getting one random number implies:
466 //*-* - Generating a random number between 0 and 1 (say r1)
467 //*-* - Look in which bin in the normalized integral r1 corresponds to
468 //*-* - Fill histogram channel
469 //*-* ntimes random numbers are generated
470 //*-*
471 //*-* One can also call TF2::GetRandom2 to get a random variate from a function.
472 //*-*
473 //*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-**-*-*-*-*-*-*-*
474
475 //*-*- Search for fname in the list of ROOT defined functions
476 TObject* fobj = gROOT->GetFunction(fname);
477 if(fobj == nullptr) {
478 Error("FillRandom", "Unknown function: %s", fname);
479 return;
480 }
481 TF2* f1 = static_cast<TF2*>(fobj);
482 if(f1 == nullptr) {
483 Error("FillRandom", "Function: %s is not a TF2", fname);
484 return;
485 }
486
487 //*-*- Allocate temporary space to store the integral and compute integral
488 Int_t nbinsx = GetNbinsX();
489 Int_t nbinsy = GetNbinsY();
490 Int_t nbins = nbinsx * nbinsy;
491
492 auto* integral = new Double_t[nbins + 1];
493 Int_t ibin = 0;
494 integral[ibin] = 0;
495 for(Int_t biny = 1; biny <= nbinsy; ++biny) {
496 for(Int_t binx = 1; binx <= nbinsx; ++binx) {
497 ++ibin;
498 Double_t fint = f1->Integral(fXaxis.GetBinLowEdge(binx), fXaxis.GetBinUpEdge(binx), fYaxis.GetBinLowEdge(biny),
499 fYaxis.GetBinUpEdge(biny));
500 integral[ibin] = integral[ibin - 1] + fint;
501 }
502 }
503
504 //*-*- Normalize integral to 1
505 if(integral[nbins] == 0) {
506 delete[] integral;
507 Error("FillRandom", "Integral = zero");
508 return;
509 }
510 for(Int_t bin = 1; bin <= nbins; ++bin) {
511 integral[bin] /= integral[nbins];
512 }
513
514 //*-*--------------Start main loop ntimes
515 for(int loop = 0; loop < ntimes; ++loop) {
516 Double_t r1 = (rng != nullptr) ? rng->Rndm(loop) : gRandom->Rndm(loop);
517 ibin = TMath::BinarySearch(nbins, &integral[0], r1);
518 Int_t biny = ibin / nbinsx;
519 Int_t binx = 1 + ibin - nbinsx * biny;
520 ++biny;
521 Fill(fXaxis.GetBinCenter(binx), fYaxis.GetBinCenter(biny));
522 }
523 delete[] integral;
524}

638void GHSym::FitSlices(TF1* f1, Int_t firstbin, Int_t lastbin, Int_t cut, Option_t* option, TObjArray* arr)
639{
640 // Project slices along X in case of a 2-D histogram, then fit each slice
641 // with function f1 and make a histogram for each fit parameter
642 // Only bins along Y between firstbin and lastbin are considered.
643 // By default (firstbin == 0, lastbin == -1), all bins in y including
644 // over- and underflows are taken into account.
645 // If f1=0, a gaussian is assumed
646 // Before invoking this function, one can set a subrange to be fitted along X
647 // via f1->SetRange(xmin,xmax)
648 // The argument option (default="QNR") can be used to change the fit options.
649 // "Q" means Quiet mode
650 // "N" means do not show the result of the fit
651 // "R" means fit the function in the specified function range
652 // "G2" merge 2 consecutive bins along X
653 // "G3" merge 3 consecutive bins along X
654 // "G4" merge 4 consecutive bins along X
655 // "G5" merge 5 consecutive bins along X
656 // "S" sliding merge: merge n consecutive bins along X accordingly to what Gn is given.
657 // It makes sense when used together with a Gn option
658 //
659 // The generated histograms are returned by adding them to arr, if arr is not NULL.
660 // arr's SetOwner() is called, to signal that it is the user's respponsability to
661 // delete the histograms, possibly by deleting the arrary.
662 // TObjArray aSlices;
663 // h2->FitSlicesX(func, 0, -1, 0, "QNR", &aSlices);
664 // will already delete the histograms once aSlice goes out of scope. aSlices will
665 // contain the histogram for the i-th parameter of the fit function at aSlices[i];
666 // aSlices[n] (n being the number of parameters) contains the chi2 distribution of
667 // the fits.
668 //
669 // If arr is NULL, the generated histograms are added to the list of objects
670 // in the current directory. It is the user's responsability to delete
671 // these histograms.
672 //
673 // Example: Assume a 2-d histogram h2
674 // Root > h2->FitSlicesX(); produces 4 TH1D histograms
675 // with h2_0 containing parameter 0(Constant) for a Gaus fit
676 // of each bin in Y projected along X
677 // with h2_1 containing parameter 1(Mean) for a gaus fit
678 // with h2_2 containing parameter 2(RMS) for a gaus fit
679 // with h2_chi2 containing the chisquare/number of degrees of freedom for a gaus fit
680 //
681 // Root > h2->FitSlicesX(0,15,22,10);
682 // same as above, but only for bins 15 to 22 along Y
683 // and only for bins in Y for which the corresponding projection
684 // along X has more than cut bins filled.
685 //
686 // NOTE: To access the generated histograms in the current directory, do eg:
687 // TH1D *h2_1 = (TH1D*)gDirectory->Get("h2_1");
688
689 Int_t nbins = fYaxis.GetNbins();
690 if(firstbin < 0) {
691 firstbin = 0;
692 }
693 if(lastbin < 0 || lastbin > nbins + 1) {
694 lastbin = nbins + 1;
695 }
696 if(lastbin < firstbin) {
697 firstbin = 0;
698 lastbin = nbins + 1;
699 }
700 TString opt = option;
701 opt.ToLower();
702 Int_t ngroup = 1;
703 if(opt.Contains("g2")) {
704 ngroup = 2;
705 opt.ReplaceAll("g2", "");
706 }
707 if(opt.Contains("g3")) {
708 ngroup = 3;
709 opt.ReplaceAll("g3", "");
710 }
711 if(opt.Contains("g4")) {
712 ngroup = 4;
713 opt.ReplaceAll("g4", "");
714 }
715 if(opt.Contains("g5")) {
716 ngroup = 5;
717 opt.ReplaceAll("g5", "");
718 }
719
720 // implement option S sliding merge for each bin using in conjunction with a given Gn
721 Int_t nstep = ngroup;
722 if(opt.Contains("s")) {
723 nstep = 1;
724 }
725
726 // default is to fit with a gaussian
727 if(f1 == nullptr) {
728 f1 = static_cast<TF1*>(gROOT->GetFunction("gaus"));
729 if(f1 == nullptr) {
730 f1 = new TF1("gaus", "gaus", fXaxis.GetXmin(), fXaxis.GetXmax());
731 } else {
732 f1->SetRange(fXaxis.GetXmin(), fXaxis.GetXmax());
733 }
734 }
735 Int_t npar = f1->GetNpar();
736 if(npar <= 0) {
737 return;
738 }
739 auto* parsave = new Double_t[npar];
740 f1->GetParameters(parsave);
741
742 if(arr != nullptr) {
743 arr->SetOwner();
744 arr->Expand(npar + 1);
745 }
746
747 // Create one histogram for each function parameter
748 auto** hlist = new TH1D*[npar];
749 auto* name = new char[2000];
750 auto* title = new char[2000];
751 const TArrayD* bins = fYaxis.GetXbins();
752 for(Int_t ipar = 0; ipar < npar; ++ipar) {
753 snprintf(name, 2000, "%s_%d", GetName(), ipar);
754 snprintf(title, 2000, "Fitted value of par[%d]=%s", ipar, f1->GetParName(ipar));
755 delete gDirectory->FindObject(name);
756 if(bins->fN == 0) {
757 hlist[ipar] = new TH1D(name, title, nbins, fYaxis.GetXmin(), fYaxis.GetXmax());
758 } else {
759 hlist[ipar] = new TH1D(name, title, nbins, bins->fArray);
760 }
761 hlist[ipar]->GetXaxis()->SetTitle(fYaxis.GetTitle());
762 if(arr != nullptr) {
763 (*arr)[ipar] = hlist[ipar];
764 }
765 }
766 snprintf(name, 2000, "%s_chi2", GetName());
767 delete gDirectory->FindObject(name);
768 TH1D* hchi2 = nullptr;
769 if(bins->fN == 0) {
770 hchi2 = new TH1D(name, "chisquare", nbins, fYaxis.GetXmin(), fYaxis.GetXmax());
771 } else {
772 hchi2 = new TH1D(name, "chisquare", nbins, bins->fArray);
773 }
774 hchi2->GetXaxis()->SetTitle(fYaxis.GetTitle());
775 if(arr != nullptr) {
776 (*arr)[npar] = hchi2;
777 }
778
779 // Loop on all bins in Y, generate a projection along X
780 // in case of sliding merge nstep=1, i.e. do slices starting for every bin
781 // now do not slices case with overflow (makes more sense)
782 for(Int_t bin = firstbin; bin + ngroup - 1 <= lastbin; bin += nstep) {
783 TH1D* hp = Projection("_temp", bin, bin + ngroup - 1, "e");
784 if(hp == nullptr) {
785 continue;
786 }
787 auto nentries = static_cast<Long64_t>(hp->GetEntries());
788 if(nentries == 0 || nentries < cut) {
789 delete hp;
790 continue;
791 }
792 f1->SetParameters(parsave);
793 hp->Fit(f1, opt.Data());
794 Int_t npfits = f1->GetNumberFitPoints();
795 if(npfits > npar && npfits >= cut) {
796 Int_t binOn = bin + ngroup / 2;
797 for(Int_t ipar = 0; ipar < npar; ++ipar) {
798 hlist[ipar]->Fill(fYaxis.GetBinCenter(binOn), f1->GetParameter(ipar));
799 hlist[ipar]->SetBinError(binOn, f1->GetParError(ipar));
800 }
801 hchi2->Fill(fYaxis.GetBinCenter(binOn), f1->GetChisquare() / (npfits - npar));
802 }
803 delete hp;
804 }
805 delete[] parsave;
806 delete[] name;
807 delete[] title;
808 delete[] hlist;
809}

832Double_t GHSym::GetBinWithContent2(Double_t c, Int_t& binx, Int_t& biny, Int_t firstxbin, Int_t lastxbin,
833 Int_t firstybin, Int_t lastybin, Double_t maxdiff) const
834{
835 // compute first cell (binx,biny) in the range [firstxbin,lastxbin][firstybin,lastybin] for which
836 // diff = abs(cell_content-c) <= maxdiff
837 // In case several cells in the specified range with diff=0 are found
838 // the first cell found is returned in binx,biny.
839 // In case several cells in the specified range satisfy diff <=maxdiff
840 // the cell with the smallest difference is returned in binx,biny.
841 // In all cases the function returns the smallest difference.
842 //
843 // NOTE1: if firstxbin < 0, firstxbin is set to 1
844 // if(lastxbin < firstxbin then lastxbin is set to the number of bins in X
845 // ie if firstxbin=1 and lastxbin=0 (default) the search is on all bins in X except
846 // for X's under- and overflow bins.
847 // if firstybin < 0, firstybin is set to 1
848 // if(lastybin < firstybin then lastybin is set to the number of bins in Y
849 // ie if firstybin=1 and lastybin=0 (default) the search is on all bins in Y except
850 // for Y's under- and overflow bins.
851 // NOTE2: if maxdiff=0 (default), the first cell with content=c is returned.
852
853 if(fDimension != 2) {
854 binx = -1;
855 biny = -1;
856 Error("GetBinWithContent2", "function is only valid for 2-D histograms");
857 return 0;
858 }
859 if(firstxbin < 0) {
860 firstxbin = 1;
861 }
862 if(lastxbin < firstxbin) {
863 lastxbin = fXaxis.GetNbins();
864 }
865 if(firstybin < 0) {
866 firstybin = 1;
867 }
868 if(lastybin < firstybin) {
869 lastybin = fYaxis.GetNbins();
870 }
871 Double_t curmax = 1e240;
872 for(Int_t j = firstybin; j <= lastybin; j++) {
873 for(Int_t i = firstxbin; i <= lastxbin; i++) {
874 Double_t diff = TMath::Abs(GetBinContent(i, j) - c);
875 if(diff <= 0) {
876 binx = i;
877 biny = j;
878 return diff;
879 }
880 if(diff < curmax && diff <= maxdiff) {
881 curmax = diff, binx = i;
882 biny = j;
883 }
884 }
885 }
886 return curmax;
887}

921Double_t GHSym::GetCovariance(Int_t axis1, Int_t axis2) const
922{
923 //*-*-*-*-*-*-*-*Return covariance between axis1 and axis2*-*-*-*-*
924 //*-* ====================================================
925
926 if(axis1 < 1 || axis2 < 1 || axis1 > 2 || axis2 > 2) {
927 Error("GetCovariance", "Wrong parameters");
928 return 0;
929 }
930 std::array<Double_t, kNstat> stats;
931 GetStats(stats.data());
932 Double_t sumw = stats[0];
933 // Double_t sumw2 = stats[1];
934 Double_t sumwx = stats[2];
935 Double_t sumwx2 = stats[3];
936 Double_t sumwy = stats[4];
937 Double_t sumwy2 = stats[5];
938 Double_t sumwxy = stats[6];
939
940 if(sumw == 0) {
941 return 0;
942 }
943 if(axis1 == 1 && axis2 == 1) {
944 return TMath::Abs(sumwx2 / sumw - sumwx / sumw * sumwx / sumw);
945 }
946 if(axis1 == 2 && axis2 == 2) {
947 return TMath::Abs(sumwy2 / sumw - sumwy / sumw * sumwy / sumw);
948 }
949 return sumwxy / sumw - sumwx / sumw * sumwy / sumw;
950}

952void GHSym::GetRandom2(Double_t& x, Double_t& y)
953{
954 // return 2 random numbers along axis x and y distributed according
955 // the cellcontents of a 2-dim histogram
956 // return a NaN if the histogram has a bin with negative content
957
958 Int_t nbinsx = GetNbinsX();
959 Int_t nbinsy = GetNbinsY();
960 Int_t nbins = nbinsx * nbinsy;
961 Double_t integral;
962 // compute integral checking that all bins have positive content (see ROOT-5894)
963 if(fIntegral != nullptr) {
964 if(fIntegral[nbins + 1] != fEntries) {
965 integral = ComputeIntegral(true);
966 } else {
967 integral = fIntegral[nbins];
968 }
969 } else {
970 integral = ComputeIntegral(true);
971 }
972 if(integral == 0) {
973 x = 0;
974 y = 0;
975 return;
976 }
977 // case histogram has negative bins
978 if(integral == TMath::QuietNaN()) {
979 x = TMath::QuietNaN();
980 y = TMath::QuietNaN();
981 return;
982 }
983
984 Double_t r1 = gRandom->Rndm();
985 Int_t ibin = TMath::BinarySearch(nbins, fIntegral, r1);
986 Int_t biny = ibin / nbinsx;
987 Int_t binx = ibin - nbinsx * biny;
988 x = fXaxis.GetBinLowEdge(binx + 1);
989 if(r1 > fIntegral[ibin]) {
990 x += fXaxis.GetBinWidth(binx + 1) * (r1 - fIntegral[ibin]) / (fIntegral[ibin + 1] - fIntegral[ibin]);
991 }
992 y = fYaxis.GetBinLowEdge(biny + 1) + fYaxis.GetBinWidth(biny + 1) * gRandom->Rndm();
993}

995void GHSym::GetStats(Double_t* stats) const
996{
997 // fill the array stats from the contents of this histogram
998 // The array stats must be correctly dimensionned in the calling program.
999 // stats[0] = sumw
1000 // stats[1] = sumw2
1001 // stats[2] = sumwx
1002 // stats[3] = sumwx2
1003 // stats[4] = sumwy
1004 // stats[5] = sumwy2
1005 // stats[6] = sumwxy
1006 //
1007 // If no axis-subranges are specified (via TAxis::SetRange), the array stats
1008 // is simply a copy of the statistics quantities computed at filling time.
1009 // If sub-ranges are specified, the function recomputes these quantities
1010 // from the bin contents in the current axis ranges.
1011 //
1012 // Note that the mean value/RMS is computed using the bins in the currently
1013 // defined ranges (see TAxis::SetRange). By default the ranges include
1014 // all bins from 1 to nbins included, excluding underflows and overflows.
1015 // To force the underflows and overflows in the computation, one must
1016 // call the static function TH1::StatOverflows(kTRUE) before filling
1017 // the histogram.
1018
1019 if(fBuffer != nullptr) {
1020 const_cast<GHSym*>(this)->BufferEmpty();
1021 }
1022
1023 Int_t bin, binx, biny;
1024 Double_t w, err;
1025 Double_t x, y;
1026 if((fTsumw == 0 && fEntries > 0) || fXaxis.TestBit(TAxis::kAxisRange) || fYaxis.TestBit(TAxis::kAxisRange)) {
1027 for(bin = 0; bin < 7; ++bin) {
1028 stats[bin] = 0;
1029 }
1030
1031 Int_t firstBinX = fXaxis.GetFirst();
1032 Int_t lastBinX = fXaxis.GetLast();
1033 Int_t firstBinY = fYaxis.GetFirst();
1034 Int_t lastBinY = fYaxis.GetLast();
1035 // include underflow/overflow if TH1::StatOverflows(kTRUE) in case no range is set on the axis
1036 if(fgStatOverflows) {
1037 if(!fXaxis.TestBit(TAxis::kAxisRange)) {
1038 if(firstBinX == 1) {
1039 firstBinX = 0;
1040 }
1041 if(lastBinX == fXaxis.GetNbins()) {
1042 lastBinX += 1;
1043 }
1044 }
1045 if(!fYaxis.TestBit(TAxis::kAxisRange)) {
1046 if(firstBinY == 1) {
1047 firstBinY = 0;
1048 }
1049 if(lastBinY == fYaxis.GetNbins()) {
1050 lastBinY += 1;
1051 }
1052 }
1053 }
1054 for(biny = firstBinY; biny <= lastBinY; ++biny) {
1055 y = fYaxis.GetBinCenter(biny);
1056 for(binx = firstBinX; binx <= lastBinX; ++binx) {
1057 bin = GetBin(binx, biny);
1058 x = fXaxis.GetBinCenter(binx);
1059 w = GetBinContent(bin);
1060 err = TMath::Abs(GetBinError(bin));
1061 stats[0] += w;
1062 stats[1] += err * err;
1063 stats[2] += w * x;
1064 stats[3] += w * x * x;
1065 stats[4] += w * y;
1066 stats[5] += w * y * y;
1067 stats[6] += w * x * y;
1068 }
1069 }
1070 } else {
1071 stats[0] = fTsumw;
1072 stats[1] = fTsumw2;
1073 stats[2] = fTsumwx;
1074 stats[3] = fTsumwx2;
1075 stats[4] = fTsumwy;
1076 stats[5] = fTsumwy2;
1077 stats[6] = fTsumwxy;
1078 }
1079}

1126#if ROOT_VERSION_CODE < ROOT_VERSION(6, 20, 0)
1127Double_t GHSym::Interpolate(Double_t x, Double_t y)
1128#else
1129Double_t GHSym::Interpolate(Double_t x, Double_t y) const
1130#endif
1131{
1132 // Given a point P(x,y), Interpolate approximates the value via bilinear
1133 // interpolation based on the four nearest bin centers
1134 // see Wikipedia, Bilinear Interpolation
1135 // Andy Mastbaum 10/8/2008
1136 // vaguely based on R.Raja 6-Sep-2008
1137
1138 Double_t f = 0;
1139 Double_t x1 = 0;
1140 Double_t x2 = 0;
1141 Double_t y1 = 0;
1142 Double_t y2 = 0;
1143 Double_t dx, dy;
1144 Int_t bin_x = fXaxis.FindBin(x);
1145 Int_t bin_y = fYaxis.FindBin(y);
1146 if(bin_x < 1 || bin_x > GetNbinsX() || bin_y < 1 || bin_y > GetNbinsY()) {
1147 Error("Interpolate", "Cannot interpolate outside histogram domain.");
1148 return 0;
1149 }
1150 Int_t quadrant = 0; // CCW from UR 1,2,3,4
1151 // which quadrant of the bin (bin_P) are we in?
1152 dx = fXaxis.GetBinUpEdge(bin_x) - x;
1153 dy = fYaxis.GetBinUpEdge(bin_y) - y;
1154 if(dx <= fXaxis.GetBinWidth(bin_x) / 2 && dy <= fYaxis.GetBinWidth(bin_y) / 2) {
1155 quadrant = 1; // upper right
1156 }
1157 if(dx > fXaxis.GetBinWidth(bin_x) / 2 && dy <= fYaxis.GetBinWidth(bin_y) / 2) {
1158 quadrant = 2; // upper left
1159 }
1160 if(dx > fXaxis.GetBinWidth(bin_x) / 2 && dy > fYaxis.GetBinWidth(bin_y) / 2) {
1161 quadrant = 3; // lower left
1162 }
1163 if(dx <= fXaxis.GetBinWidth(bin_x) / 2 && dy > fYaxis.GetBinWidth(bin_y) / 2) {
1164 quadrant = 4; // lower right
1165 }
1166 switch(quadrant) {
1167 case 1:
1168 x1 = fXaxis.GetBinCenter(bin_x);
1169 y1 = fYaxis.GetBinCenter(bin_y);
1170 x2 = fXaxis.GetBinCenter(bin_x + 1);
1171 y2 = fYaxis.GetBinCenter(bin_y + 1);
1172 break;
1173 case 2:
1174 x1 = fXaxis.GetBinCenter(bin_x - 1);
1175 y1 = fYaxis.GetBinCenter(bin_y);
1176 x2 = fXaxis.GetBinCenter(bin_x);
1177 y2 = fYaxis.GetBinCenter(bin_y + 1);
1178 break;
1179 case 3:
1180 x1 = fXaxis.GetBinCenter(bin_x - 1);
1181 y1 = fYaxis.GetBinCenter(bin_y - 1);
1182 x2 = fXaxis.GetBinCenter(bin_x);
1183 y2 = fYaxis.GetBinCenter(bin_y);
1184 break;
1185 case 4:
1186 x1 = fXaxis.GetBinCenter(bin_x);
1187 y1 = fYaxis.GetBinCenter(bin_y - 1);
1188 x2 = fXaxis.GetBinCenter(bin_x + 1);
1189 y2 = fYaxis.GetBinCenter(bin_y);
1190 break;
1191 }
1192 Int_t bin_x1 = fXaxis.FindBin(x1);
1193 if(bin_x1 < 1) {
1194 bin_x1 = 1;
1195 }
1196 Int_t bin_x2 = fXaxis.FindBin(x2);
1197 if(bin_x2 > GetNbinsX()) {
1198 bin_x2 = GetNbinsX();
1199 }
1200 Int_t bin_y1 = fYaxis.FindBin(y1);
1201 if(bin_y1 < 1) {
1202 bin_y1 = 1;
1203 }
1204 Int_t bin_y2 = fYaxis.FindBin(y2);
1205 if(bin_y2 > GetNbinsY()) {
1206 bin_y2 = GetNbinsY();
1207 }
1208 Int_t bin_q22 = GetBin(bin_x2, bin_y2);
1209 Int_t bin_q12 = GetBin(bin_x1, bin_y2);
1210 Int_t bin_q11 = GetBin(bin_x1, bin_y1);
1211 Int_t bin_q21 = GetBin(bin_x2, bin_y1);
1212 Double_t q11 = GetBinContent(bin_q11);
1213 Double_t q12 = GetBinContent(bin_q12);
1214 Double_t q21 = GetBinContent(bin_q21);
1215 Double_t q22 = GetBinContent(bin_q22);
1216 Double_t d = 1.0 * (x2 - x1) * (y2 - y1);
1217 f = 1.0 * q11 / d * (x2 - x) * (y2 - y) + 1.0 * q21 / d * (x - x1) * (y2 - y) + 1.0 * q12 / d * (x2 - x) * (y - y1) +
1218 1.0 * q22 / d * (x - x1) * (y - y1);
1219 return f;
1220}

1233Double_t GHSym::KolmogorovTest(const TH1* h2, Option_t* option) const
1234{
1235 // Statistical test of compatibility in shape between
1236 // THIS histogram and h2, using Kolmogorov test.
1237 // Default: Ignore under- and overflow bins in comparison
1238 //
1239 // option is a character string to specify options
1240 // "U" include Underflows in test
1241 // "O" include Overflows
1242 // "N" include comparison of normalizations
1243 // "D" Put out a line of "Debug" printout
1244 // "M" Return the Maximum Kolmogorov distance instead of prob
1245 //
1246 // The returned function value is the probability of test
1247 // (much less than one means NOT compatible)
1248 //
1249 // The KS test uses the distance between the pseudo-CDF's obtained
1250 // from the histogram. Since in 2D the order for generating the pseudo-CDF is
1251 // arbitrary, two pairs of pseudo-CDF are used, one starting from the x axis the
1252 // other from the y axis and the maximum distance is the average of the two maximum
1253 // distances obtained.
1254 //
1255 // Code adapted by Rene Brun from original HBOOK routine HDIFF
1256
1257 TString opt = option;
1258 opt.ToUpper();
1259
1260 Double_t prb = 0;
1261 TH1* h1 = const_cast<TH1*>(static_cast<const TH1*>(this));
1262 if(h2 == nullptr) {
1263 return 0;
1264 }
1265 TAxis* xaxis1 = h1->GetXaxis();
1266 TAxis* xaxis2 = const_cast<TAxis*>(h2->GetXaxis());
1267 TAxis* yaxis1 = h1->GetYaxis();
1268 TAxis* yaxis2 = const_cast<TAxis*>(h2->GetYaxis());
1269 Int_t ncx1 = xaxis1->GetNbins();
1270 Int_t ncx2 = xaxis2->GetNbins();
1271 Int_t ncy1 = yaxis1->GetNbins();
1272 Int_t ncy2 = yaxis2->GetNbins();
1273
1274 // Check consistency of dimensions
1275 if(h1->GetDimension() != 2 || h2->GetDimension() != 2) {
1276 Error("KolmogorovTest", "Histograms must be 2-D\n");
1277 return 0;
1278 }
1279
1280 // Check consistency in number of channels
1281 if(ncx1 != ncx2) {
1282 Error("KolmogorovTest", "Number of channels in X is different, %d and %d\n", ncx1, ncx2);
1283 return 0;
1284 }
1285 if(ncy1 != ncy2) {
1286 Error("KolmogorovTest", "Number of channels in Y is different, %d and %d\n", ncy1, ncy2);
1287 return 0;
1288 }
1289
1290 // Check consistency in channel edges
1291 Bool_t afunc1 = kFALSE;
1292 Bool_t afunc2 = kFALSE;
1293 Double_t difprec = 1e-5;
1294 Double_t diff1 = TMath::Abs(xaxis1->GetXmin() - xaxis2->GetXmin());
1295 Double_t diff2 = TMath::Abs(xaxis1->GetXmax() - xaxis2->GetXmax());
1296 if(diff1 > difprec || diff2 > difprec) {
1297 Error("KolmogorovTest", "histograms with different binning along X");
1298 return 0;
1299 }
1300 diff1 = TMath::Abs(yaxis1->GetXmin() - yaxis2->GetXmin());
1301 diff2 = TMath::Abs(yaxis1->GetXmax() - yaxis2->GetXmax());
1302 if(diff1 > difprec || diff2 > difprec) {
1303 Error("KolmogorovTest", "histograms with different binning along Y");
1304 return 0;
1305 }
1306
1307 // Should we include Uflows, Oflows?
1308 Int_t ibeg = 1;
1309 Int_t jbeg = 1;
1310 Int_t iend = ncx1;
1311 Int_t jend = ncy1;
1312 if(opt.Contains("U")) {
1313 ibeg = 0;
1314 jbeg = 0;
1315 }
1316 if(opt.Contains("O")) {
1317 iend = ncx1 + 1;
1318 jend = ncy1 + 1;
1319 }
1320
1321 Int_t i, j;
1322 Double_t sum1 = 0;
1323 Double_t sum2 = 0;
1324 Double_t w1 = 0;
1325 Double_t w2 = 0;
1326 for(i = ibeg; i <= iend; ++i) {
1327 for(j = jbeg; j <= jend; ++j) {
1328 sum1 += h1->GetBinContent(i, j);
1329 sum2 += h2->GetBinContent(i, j);
1330 Double_t ew1 = h1->GetBinError(i, j);
1331 Double_t ew2 = h2->GetBinError(i, j);
1332 w1 += ew1 * ew1;
1333 w2 += ew2 * ew2;
1334 }
1335 }
1336
1337 // Check that both scatterplots contain events
1338 if(sum1 == 0) {
1339 Error("KolmogorovTest", "Integral is zero for h1=%s\n", h1->GetName());
1340 return 0;
1341 }
1342 if(sum2 == 0) {
1343 Error("KolmogorovTest", "Integral is zero for h2=%s\n", h2->GetName());
1344 return 0;
1345 }
1346 // calculate the effective entries.
1347 // the case when errors are zero (w1 == 0 or w2 ==0) are equivalent to
1348 // compare to a function. In that case the rescaling is done only on sqrt(esum2) or sqrt(esum1)
1349 Double_t esum1 = 0, esum2 = 0;
1350 if(w1 > 0) {
1351 esum1 = sum1 * sum1 / w1;
1352 } else {
1353 afunc1 = kTRUE; // use later for calculating z
1354 }
1355
1356 if(w2 > 0) {
1357 esum2 = sum2 * sum2 / w2;
1358 } else {
1359 afunc2 = kTRUE; // use later for calculating z
1360 }
1361
1362 if(afunc2 && afunc1) {
1363 Error("KolmogorovTest", "Errors are zero for both histograms\n");
1364 return 0;
1365 }
1366
1367 // Find first Kolmogorov distance
1368 Double_t s1 = 1 / sum1;
1369 Double_t s2 = 1 / sum2;
1370 Double_t dfmax1 = 0;
1371 Double_t rsum1 = 0, rsum2 = 0;
1372 for(i = ibeg; i <= iend; ++i) {
1373 for(j = jbeg; j <= jend; ++j) {
1374 rsum1 += s1 * h1->GetCellContent(i, j);
1375 rsum2 += s2 * h2->GetCellContent(i, j);
1376 dfmax1 = TMath::Max(dfmax1, TMath::Abs(rsum1 - rsum2));
1377 }
1378 }
1379
1380 // Find second Kolmogorov distance
1381 Double_t dfmax2 = 0;
1382 rsum1 = 0, rsum2 = 0;
1383 for(j = jbeg; j <= jend; ++j) {
1384 for(i = ibeg; i <= iend; ++i) {
1385 rsum1 += s1 * h1->GetCellContent(i, j);
1386 rsum2 += s2 * h2->GetCellContent(i, j);
1387 dfmax2 = TMath::Max(dfmax2, TMath::Abs(rsum1 - rsum2));
1388 }
1389 }
1390
1391 // Get Kolmogorov probability: use effective entries, esum1 or esum2, for normalizing it
1392 Double_t factnm;
1393 if(afunc1) {
1394 factnm = TMath::Sqrt(esum2);
1395 } else if(afunc2) {
1396 factnm = TMath::Sqrt(esum1);
1397 } else {
1398 factnm = TMath::Sqrt(esum1 * sum2 / (esum1 + esum2));
1399 }
1400
1401 // take average of the two distances
1402 Double_t dfmax = 0.5 * (dfmax1 + dfmax2);
1403 Double_t z = dfmax * factnm;
1404
1405 prb = TMath::KolmogorovProb(z);
1406
1407 Double_t prb1 = 0;
1408 Double_t prb2 = 0;
1409 // option N to combine normalization makes sense if both afunc1 and afunc2 are false
1410 if(opt.Contains("N") && !(afunc1 || afunc2)) {
1411 // Combine probabilities for shape and normalization
1412 prb1 = prb;
1413 Double_t d12 = esum1 - esum2;
1414 Double_t chi2 = d12 * d12 / (esum1 + esum2);
1415 prb2 = TMath::Prob(chi2, 1);
1416 // see Eadie et al., section 11.6.2
1417 if(prb > 0 && prb2 > 0) {
1418 prb = prb * prb2 * (1 - TMath::Log(prb * prb2));
1419 } else {
1420 prb = 0;
1421 }
1422 }
1423 // debug printout
1424 if(opt.Contains("D")) {
1425 std::cout << " Kolmo Prob h1 = " << h1->GetName() << ", sum1 = " << sum1 << std::endl;
1426 std::cout << " Kolmo Prob h2 = " << h2->GetName() << ", sum2 = " << sum2 << std::endl;
1427 std::cout << " Kolmo Probabil = " << prb << ", Max dist = " << dfmax << std::endl;
1428 if(opt.Contains("N")) {
1429 std::cout << " Kolmo Probabil = " << prb1 << " for shape alone, " << prb2 << " for normalisation alone" << std::endl;
1430 }
1431 }
1432 // This numerical error condition should never occur:
1433 if(TMath::Abs(rsum1 - 1) > 0.002) {
1434 Warning("KolmogorovTest", "Numerical problems with h1=%s\n", h1->GetName());
1435 }
1436 if(TMath::Abs(rsum2 - 1) > 0.002) {
1437 Warning("KolmogorovTest", "Numerical problems with h2=%s\n", h2->GetName());
1438 }
1439
1440 if(opt.Contains("M")) {
1441 return dfmax; // return avergae of max distance
1442 }
1443
1444 return prb;
1445}

1447Long64_t GHSym::Merge(TCollection* list)
1448{
1449 // Add all histograms in the collection to this histogram.
1450 // This function computes the min/max for the axes,
1451 // compute a new number of bins, if necessary,
1452 // add bin contents, errors and statistics.
1453 // If overflows are present and limits are different the function will fail.
1454 // The function returns the total number of entries in the result histogram
1455 // if the merge is successfull, -1 otherwise.
1456 //
1457 // IMPORTANT remark. The 2 axis x and y may have different number
1458 // of bins and different limits, BUT the largest bin width must be
1459 // a multiple of the smallest bin width and the upper limit must also
1460 // be a multiple of the bin width.
1461
1462 if(list == nullptr) {
1463 return 0;
1464 }
1465 if(list->IsEmpty()) {
1466 return static_cast<Long64_t>(GetEntries());
1467 }
1468
1469 TList inlist;
1470 inlist.AddAll(list);
1471
1472 TAxis newXAxis;
1473 TAxis newYAxis;
1474 Bool_t initialLimitsFound = kFALSE;
1475 Bool_t allSameLimits = kTRUE;
1476 Bool_t sameLimitsX = kTRUE;
1477 Bool_t sameLimitsY = kTRUE;
1478 Bool_t allHaveLimits = kTRUE;
1479 Bool_t firstHistWithLimits = kTRUE;
1480
1481 TIter next(&inlist);
1482 GHSym* h = this;
1483 do {
1484 Bool_t hasLimits = h->GetXaxis()->GetXmin() < h->GetXaxis()->GetXmax();
1485 allHaveLimits = allHaveLimits && hasLimits;
1486
1487 if(hasLimits) {
1488 h->BufferEmpty();
1489
1490 // this is done in case the first histograms are empty and
1491 // the histogram have different limits
1492 if(firstHistWithLimits) {
1493 // set axis limits in the case the first histogram did not have limits
1494 if(h != this) {
1495 if(!SameLimitsAndNBins(fXaxis, *(h->GetXaxis()))) {
1496 if(h->GetXaxis()->GetXbins()->GetSize() != 0) {
1497 fXaxis.Set(h->GetXaxis()->GetNbins(), h->GetXaxis()->GetXbins()->GetArray());
1498 } else {
1499 fXaxis.Set(h->GetXaxis()->GetNbins(), h->GetXaxis()->GetXmin(), h->GetXaxis()->GetXmax());
1500 }
1501 }
1502 if(!SameLimitsAndNBins(fYaxis, *(h->GetYaxis()))) {
1503 if(h->GetYaxis()->GetXbins()->GetSize() != 0) {
1504 fYaxis.Set(h->GetYaxis()->GetNbins(), h->GetYaxis()->GetXbins()->GetArray());
1505 } else {
1506 fYaxis.Set(h->GetYaxis()->GetNbins(), h->GetYaxis()->GetXmin(), h->GetYaxis()->GetXmax());
1507 }
1508 }
1509 }
1510 firstHistWithLimits = kFALSE;
1511 }
1512
1513 if(!initialLimitsFound) {
1514 // this is executed the first time an histogram with limits is found
1515 // to set some initial values on the new axes
1516 initialLimitsFound = kTRUE;
1517 if(h->GetXaxis()->GetXbins()->GetSize() != 0) {
1518 newXAxis.Set(h->GetXaxis()->GetNbins(), h->GetXaxis()->GetXbins()->GetArray());
1519 } else {
1520 newXAxis.Set(h->GetXaxis()->GetNbins(), h->GetXaxis()->GetXmin(), h->GetXaxis()->GetXmax());
1521 }
1522 if(h->GetYaxis()->GetXbins()->GetSize() != 0) {
1523 newYAxis.Set(h->GetYaxis()->GetNbins(), h->GetYaxis()->GetXbins()->GetArray());
1524 } else {
1525 newYAxis.Set(h->GetYaxis()->GetNbins(), h->GetYaxis()->GetXmin(), h->GetYaxis()->GetXmax());
1526 }
1527 } else {
1528 // check first if histograms have same bins in X
1529 if(!SameLimitsAndNBins(newXAxis, *(h->GetXaxis()))) {
1530 sameLimitsX = kFALSE;
1531 // recompute in this case the optimal limits
1532 // The condition to works is that the histogram have same bin with
1533 // and one common bin edge
1534 if(!RecomputeAxisLimits(newXAxis, *(h->GetXaxis()))) {
1535 Error("Merge", "Cannot merge histograms - limits are inconsistent:\n "
1536 "first: (%d, %f, %f), second: (%d, %f, %f)",
1537 newXAxis.GetNbins(), newXAxis.GetXmin(), newXAxis.GetXmax(), h->GetXaxis()->GetNbins(),
1538 h->GetXaxis()->GetXmin(), h->GetXaxis()->GetXmax());
1539 return -1;
1540 }
1541 }
1542
1543 // check first if histograms have same bins in Y
1544 if(!SameLimitsAndNBins(newYAxis, *(h->GetYaxis()))) {
1545 sameLimitsY = kFALSE;
1546 // recompute in this case the optimal limits
1547 // The condition to works is that the histogram have same bin with
1548 // and one common bin edge
1549 if(!RecomputeAxisLimits(newYAxis, *(h->GetYaxis()))) {
1550 Error("Merge", "Cannot merge histograms - limits are inconsistent:\n "
1551 "first: (%d, %f, %f), second: (%d, %f, %f)",
1552 newYAxis.GetNbins(), newYAxis.GetXmin(), newYAxis.GetXmax(), h->GetYaxis()->GetNbins(),
1553 h->GetYaxis()->GetXmin(), h->GetYaxis()->GetXmax());
1554 return -1;
1555 }
1556 }
1557 allSameLimits = sameLimitsY && sameLimitsX;
1558 }
1559 }
1560 } while((h = static_cast<GHSym*>(next())) != nullptr);
1561 if(h == nullptr && ((*next) != nullptr)) {
1562 Error("Merge", "Attempt to merge object of class: %s to a %s", (*next)->ClassName(), ClassName());
1563 return -1;
1564 }
1565 next.Reset();
1566
1567 // In the case of histogram with different limits
1568 // newX(Y)Axis will now have the new found limits
1569 // but one needs first to clone this histogram to perform the merge
1570 // The clone is not needed when all histograms have the same limits
1571 TH2* hclone = nullptr;
1572 if(!allSameLimits) {
1573 // We don't want to add the clone to gDirectory,
1574 // so remove our kMustCleanup bit temporarily
1575 Bool_t mustCleanup = TestBit(kMustCleanup);
1576 if(mustCleanup) {
1577 ResetBit(kMustCleanup);
1578 }
1579 hclone = static_cast<TH2*>(IsA()->New());
1580 hclone->SetDirectory(nullptr);
1581 Copy(*hclone);
1582 if(mustCleanup) {
1583 SetBit(kMustCleanup);
1584 }
1585 BufferEmpty(1); // To remove buffer.
1586 Reset(); // BufferEmpty sets limits so we can't use it later.
1587 SetEntries(0);
1588 inlist.AddFirst(hclone);
1589 }
1590
1591 if(!allSameLimits && initialLimitsFound) {
1592 if(!sameLimitsX) {
1593 fXaxis.SetRange(0, 0);
1594 if(newXAxis.GetXbins()->GetSize() != 0) {
1595 fXaxis.Set(newXAxis.GetNbins(), newXAxis.GetXbins()->GetArray());
1596 } else {
1597 fXaxis.Set(newXAxis.GetNbins(), newXAxis.GetXmin(), newXAxis.GetXmax());
1598 }
1599 }
1600 if(!sameLimitsY) {
1601 fYaxis.SetRange(0, 0);
1602 if(newYAxis.GetXbins()->GetSize() != 0) {
1603 fYaxis.Set(newYAxis.GetNbins(), newYAxis.GetXbins()->GetArray());
1604 } else {
1605 fYaxis.Set(newYAxis.GetNbins(), newYAxis.GetXmin(), newYAxis.GetXmax());
1606 }
1607 }
1608 fZaxis.Set(1, 0, 1);
1609 fNcells = (fXaxis.GetNbins() + 2) * (fYaxis.GetNbins() + 2);
1610 SetBinsLength(fNcells);
1611 if(fSumw2.fN != 0) {
1612 fSumw2.Set(fNcells);
1613 }
1614 }
1615
1616 if(!allHaveLimits) {
1617 // fill this histogram with all the data from buffers of histograms without limits
1618 while((h = static_cast<GHSym*>(next())) != nullptr) {
1619 if(h->GetXaxis()->GetXmin() >= h->GetXaxis()->GetXmax() && (h->fBuffer != nullptr)) {
1620 // no limits
1621 Int_t nbentries = static_cast<Int_t>(h->fBuffer[0]);
1622 for(Int_t i = 0; i < nbentries; i++) {
1623 Fill(h->fBuffer[3 * i + 2], h->fBuffer[3 * i + 3], h->fBuffer[3 * i + 1]);
1624 }
1625 // Entries from buffers have to be filled one by one
1626 // because FillN doesn't resize histograms.
1627 }
1628 }
1629 if(!initialLimitsFound) {
1630 if(hclone != nullptr) {
1631 inlist.Remove(hclone);
1632 delete hclone;
1633 }
1634 return static_cast<Long64_t>(GetEntries()); // all histograms have been processed
1635 }
1636 next.Reset();
1637 }
1638
1639 // merge bin contents and errors
1640 Double_t stats[kNstat];
1641 Double_t totstats[kNstat];
1642 for(Int_t i = 0; i < kNstat; ++i) {
1643 totstats[i] = stats[i] = 0;
1644 }
1645 GetStats(totstats);
1646 Double_t nentries = GetEntries();
1647 Int_t binx, biny, ix, iy, nx, ny, ibin;
1648 Double_t cu;
1649 Bool_t canExtend = CanExtendAllAxes();
1650 SetCanExtend(TH1::kNoAxis); // reset, otherwise setting the under/overflow will extend the axis
1651
1652 while((h = static_cast<GHSym*>(next())) != nullptr) {
1653 // skip empty histograms
1654 Double_t histEntries = h->GetEntries();
1655 if(h->fTsumw == 0 && histEntries == 0) {
1656 continue;
1657 }
1658
1659 // process only if the histogram has limits; otherwise it was processed before
1660 if(h->GetXaxis()->GetXmin() < h->GetXaxis()->GetXmax()) {
1661 // import statistics
1662 h->GetStats(stats);
1663 for(Int_t i = 0; i < kNstat; ++i) {
1664 totstats[i] += stats[i];
1665 }
1666 nentries += histEntries;
1667
1668 nx = h->GetXaxis()->GetNbins();
1669 ny = h->GetYaxis()->GetNbins();
1670
1671 for(biny = 0; biny <= ny + 1; ++biny) {
1672 if(!allSameLimits) {
1673 iy = fYaxis.FindBin(h->GetYaxis()->GetBinCenter(biny));
1674 } else {
1675 iy = biny;
1676 }
1677 for(binx = 0; binx <= nx + 1; ++binx) {
1678 cu = h->GetBinContent(binx, biny);
1679 if(!allSameLimits) {
1680 if(cu != 0 && ((!sameLimitsX && (binx == 0 || binx == nx + 1)) ||
1681 (!sameLimitsY && (biny == 0 || biny == ny + 1)))) {
1682 Error("Merge", "Cannot merge histograms - the histograms have"
1683 " different limits and undeflows/overflows are present."
1684 " The initial histogram is now broken!");
1685 return -1;
1686 }
1687 ix = fXaxis.FindBin(h->GetXaxis()->GetBinCenter(binx));
1688 } else {
1689 // case histograms with the same limits
1690 ix = binx;
1691 }
1692 ibin = GetBin(ix, iy);
1693
1694 if(ibin < 0) {
1695 continue;
1696 }
1697 AddBinContent(ibin, cu);
1698 if(fSumw2.fN != 0) {
1699 Double_t error1 = h->GetBinError(GetBin(binx, biny));
1700 fSumw2.fArray[ibin] += error1 * error1;
1701 }
1702 }
1703 }
1704 }
1705 }
1706 SetCanExtend(static_cast<UInt_t>(canExtend));
1707
1708 // copy merged stats
1709 PutStats(totstats);
1710 SetEntries(nentries);
1711 if(hclone != nullptr) {
1712 inlist.Remove(hclone);
1713 delete hclone;
1714 }
1715 return static_cast<Long64_t>(nentries);
1716}

1718TProfile* GHSym::Profile(const char* name, Int_t firstbin, Int_t lastbin, Option_t* option) const
1719{
1720 // *-*-*-*-*Project a 2-D histogram into a profile histogram along X*-*-*-*-*-*
1721 // *-* ========================================================
1722 //
1723 // The projection is made from the channels along the Y axis
1724 // ranging from firstybin to lastybin included.
1725 // By default, bins 1 to ny are included
1726 // When all bins are included, the number of entries in the projection
1727 // is set to the number of entries of the 2-D histogram, otherwise
1728 // the number of entries is incremented by 1 for all non empty cells.
1729 //
1730 // if option "d" is specified, the profile is drawn in the current pad.
1731 //
1732 // if option "o" original axis range of the target axes will be
1733 // kept, but only bins inside the selected range will be filled.
1734 //
1735 // The option can also be used to specify the projected profile error type.
1736 // Values which can be used are 's', 'i', or 'g'. See TProfile::BuildOptions for details
1737 //
1738 // Using a TCutG object, it is possible to select a sub-range of a 2-D histogram.
1739 // One must create a graphical cut (mouse or C++) and specify the name
1740 // of the cut between [] in the option.
1741 // For example, with a TCutG named "cutg", one can call:
1742 // myhist->ProfileX(" ",firstybin,lastybin,"[cutg]");
1743 // To invert the cut, it is enough to put a "-" in front of its name:
1744 // myhist->ProfileX(" ",firstybin,lastybin,"[-cutg]");
1745 // It is possible to apply several cuts ("," means logical AND):
1746 // myhist->ProfileX(" ",firstybin,lastybin,"[cutg1,cutg2]");
1747 //
1748 // NOTE that if a TProfile named "name" exists in the current directory or pad with
1749 // a compatible axis the profile is reset and filled again with the projected contents of the TH2.
1750 // In the case of axis incompatibility an error is reported and a NULL pointer is returned.
1751 //
1752 // NOTE that the X axis attributes of the TH2 are copied to the X axis of the profile.
1753 //
1754 // NOTE that the default under- / overflow behavior differs from what ProjectionX
1755 // does! Profiles take the bin center into account, so here the under- and overflow
1756 // bins are ignored by default.
1757
1758 TString opt = option;
1759 // extract cut infor
1760 TString cut;
1761 Int_t i1 = opt.Index("[");
1762 if(i1 >= 0) {
1763 Int_t i2 = opt.Index("]");
1764 cut = opt(i1, i2 - i1 + 1);
1765 }
1766 opt.ToLower();
1767 bool originalRange = opt.Contains("o");
1768
1769 Int_t inN = fYaxis.GetNbins();
1770 const char* expectedName = "_pf";
1771
1772 Int_t firstOutBin, lastOutBin;
1773 firstOutBin = fXaxis.GetFirst();
1774 lastOutBin = fXaxis.GetLast();
1775 if(firstOutBin == 0 && lastOutBin == 0) {
1776 firstOutBin = 1;
1777 lastOutBin = fXaxis.GetNbins();
1778 }
1779
1780 if(lastbin < firstbin && fYaxis.TestBit(TAxis::kAxisRange)) {
1781 firstbin = fYaxis.GetFirst();
1782 lastbin = fYaxis.GetLast();
1783 // For special case of TAxis::SetRange, when first == 1 and last
1784 // = N and the range bit has been set, the TAxis will return 0
1785 // for both.
1786 if(firstbin == 0 && lastbin == 0) {
1787 firstbin = 1;
1788 lastbin = fYaxis.GetNbins();
1789 }
1790 }
1791 if(firstbin < 0) {
1792 firstbin = 1;
1793 }
1794 if(lastbin < 0) {
1795 lastbin = inN;
1796 }
1797 if(lastbin > inN + 1) {
1798 lastbin = inN;
1799 }
1800
1801 // Create the profile histogram
1802 char* pname = const_cast<char*>(name);
1803 if((name != nullptr) && strcmp(name, expectedName) == 0) {
1804 Int_t nch = strlen(GetName()) + 5;
1805 pname = new char[nch];
1806 snprintf(pname, nch, "%s%s", GetName(), name);
1807 }
1808 TProfile* h1 = nullptr;
1809 // check if a profile with identical name exist
1810 // if compatible reset and re-use previous histogram
1811 TObject* h1obj = gROOT->FindObject(pname);
1812 if((h1obj != nullptr) && h1obj->InheritsFrom(TH1::Class())) {
1813 if(h1obj->IsA() != TProfile::Class()) {
1814 Error("DoProfile", "Histogram with name %s must be a TProfile and is a %s", name, h1obj->ClassName());
1815 return nullptr;
1816 }
1817 h1 = static_cast<TProfile*>(h1obj);
1818 // reset the existing histogram and set always the new binning for the axis
1819 // This avoid problems when the histogram already exists and the histograms is rebinned or its range has changed
1820 // (see https://savannah.cern.ch/bugs/?94101 or https://savannah.cern.ch/bugs/?95808 )
1821 h1->Reset();

2415 Double_t binError0, binError2;
2416 Int_t oldxbin2, oldybin2;
2417 Int_t ufbin, ofbin;
2418 oldxbin2 = 1;
2419 for(xbin = 1; xbin <= newbins; ++xbin) {
2420 binContent0 = binContent2 = 0;
2421 binError0 = binError2 = 0;
2422 for(i = 0; i < ngroup; ++i) {
2423 if(oldxbin2 + i > nbins) {
2424 break;
2425 }
2426 // old underflow bin (in y)
2427 ufbin = oldxbin2 + i;
2428 binContent0 += oldBins[ufbin];
2429 if(oldErrors != nullptr) {
2430 binError0 += oldErrors[ufbin] * oldErrors[ufbin];
2431 }
2432 for(ybin = oldybin; ybin <= nbins + 1; ++ybin) {
2433 // old overflow bin (in y)
2434 ofbin = ufbin + ybin * (nbins + 2);
2435 binContent2 += oldBins[ofbin];
2436 if(oldErrors != nullptr) {
2437 binError2 += oldErrors[ofbin] * oldErrors[ofbin];
2438 }
2439 }
2440 }
2441 hnew->SetBinContent(xbin, 0, binContent0);
2442 hnew->SetBinContent(xbin, newbins + 1, binContent2);
2443 if(oldErrors != nullptr) {
2444 hnew->SetBinError(xbin, 0, TMath::Sqrt(binError0));
2445 hnew->SetBinError(xbin, newbins + 1, TMath::Sqrt(binError2));
2446 }
2447 oldxbin2 += ngroup;
2448 }
2449
2450 // recompute under/overflow contents in x for the new y bins
2451 oldybin2 = 1;
2452 for(ybin = 1; ybin <= newbins; ++ybin) {
2453 binContent0 = binContent2 = 0;
2454 binError0 = binError2 = 0;
2455 for(i = 0; i < ngroup; ++i) {
2456 if(oldybin2 + i > nbins) {
2457 break;
2458 }
2459 // old underflow bin (in x)
2460 ufbin = (oldybin2 + i) * (nbins + 2);
2461 binContent0 += oldBins[ufbin];
2462 if(oldErrors != nullptr) {
2463 binError0 += oldErrors[ufbin] * oldErrors[ufbin];
2464 }
2465 for(xbin = oldxbin; xbin <= nbins + 1; ++xbin) {
2466 ofbin = ufbin + xbin;
2467 binContent2 += oldBins[ofbin];
2468 if(oldErrors != nullptr) {
2469 binError2 += oldErrors[ofbin] * oldErrors[ofbin];
2470 }
2471 }
2472 }
2473 hnew->SetBinContent(0, ybin, binContent0);
2474 hnew->SetBinContent(newbins + 1, ybin, binContent2);
2475 if(oldErrors != nullptr) {
2476 hnew->SetBinError(0, ybin, TMath::Sqrt(binError0));
2477 hnew->SetBinError(newbins + 1, ybin, TMath::Sqrt(binError2));
2478 }
2479 oldybin2 += ngroup;
2480 }
2481 }
2482
2483 // Restore x axis attributes
2484 fXaxis.SetNdivisions(nXdivisions);
2485 fXaxis.SetAxisColor(xAxisColor);
2486 fXaxis.SetLabelColor(xLabelColor);
2487 fXaxis.SetLabelFont(xLabelFont);
2488 fXaxis.SetLabelOffset(xLabelOffset);
2489 fXaxis.SetLabelSize(xLabelSize);
2490 fXaxis.SetTickLength(xTickLength);
2491 fXaxis.SetTitleOffset(xTitleOffset);
2492 fXaxis.SetTitleSize(xTitleSize);
2493 fXaxis.SetTitleColor(xTitleColor);
2494 fXaxis.SetTitleFont(xTitleFont);
2495 // Restore y axis attributes
2496 fYaxis.SetNdivisions(nYdivisions);
2497 fYaxis.SetAxisColor(yAxisColor);
2498 fYaxis.SetLabelColor(yLabelColor);
2499 fYaxis.SetLabelFont(yLabelFont);
2500 fYaxis.SetLabelOffset(yLabelOffset);
2501 fYaxis.SetLabelSize(yLabelSize);
2502 fYaxis.SetTickLength(yTickLength);
2503 fYaxis.SetTitleOffset(yTitleOffset);
2504 fYaxis.SetTitleSize(yTitleSize);
2505 fYaxis.SetTitleColor(yTitleColor);
2506 fYaxis.SetTitleFont(yTitleFont);
2507
2508 // restore statistics and entries modified by SetBinContent
2509 hnew->SetEntries(entries);
2510 if(!resetStat) {
2511 hnew->PutStats(stat);
2512 }
2513
2514 delete[] oldBins;
2515 if(oldErrors != nullptr) {
2516 delete[] oldErrors;
2517 }
2518 return hnew;
2519}

2607void GHSym::Smooth(Int_t ntimes, Option_t* option)
2608{
2609 // Smooth bin contents of this 2-d histogram using kernel algorithms
2610 // similar to the ones used in the raster graphics community.
2611 // Bin contents in the active range are replaced by their smooth values.
2612 // If Errors are defined via Sumw2, they are also scaled and computed.
2613 // However, note the resulting errors will be correlated between different-bins, so
2614 // the errors should not be used blindly to perform any calculation involving several bins,
2615 // like fitting the histogram. One would need to compute also the bin by bin correlation matrix.
2616 //
2617 // 3 kernels are proposed k5a, k5b and k3a.
2618 // k5a and k5b act on 5x5 cells (i-2,i-1,i,i+1,i+2, and same for j)
2619 // k5b is a bit more stronger in smoothing
2620 // k3a acts only on 3x3 cells (i-1,i,i+1, and same for j).
2621 // By default the kernel "k5a" is used. You can select the kernels "k5b" or "k3a"
2622 // via the option argument.
2623 // If TAxis::SetRange has been called on the x or/and y axis, only the bins
2624 // in the specified range are smoothed.
2625 // In the current implementation if the first argument is not used (default value=1).
2626 //
2627 // implementation by David McKee (dmckee@bama.ua.edu). Extended by Rene Brun
2628
2629 Double_t k5a[5][5] = {{0, 0, 1, 0, 0}, {0, 2, 2, 2, 0}, {1, 2, 5, 2, 1}, {0, 2, 2, 2, 0}, {0, 0, 1, 0, 0}};
2630 Double_t k5b[5][5] = {{0, 1, 2, 1, 0}, {1, 2, 4, 2, 1}, {2, 4, 8, 4, 2}, {1, 2, 4, 2, 1}, {0, 1, 2, 1, 0}};
2631 Double_t k3a[3][3] = {{0, 1, 0}, {1, 2, 1}, {0, 1, 0}};
2632
2633 if(ntimes > 1) {
2634 Warning("Smooth", "Currently only ntimes=1 is supported");
2635 }
2636 TString opt = option;
2637 opt.ToLower();
2638 Int_t ksize_x = 5;
2639 Int_t ksize_y = 5;
2640 Double_t* kernel = &k5a[0][0];
2641 if(opt.Contains("k5b")) {
2642 kernel = &k5b[0][0];
2643 }
2644 if(opt.Contains("k3a")) {
2645 kernel = &k3a[0][0];
2646 ksize_x = 3;
2647 ksize_y = 3;
2648 }
2649
2650 // find i,j ranges
2651 Int_t ifirst = fXaxis.GetFirst();
2652 Int_t ilast = fXaxis.GetLast();
2653 Int_t jfirst = fYaxis.GetFirst();
2654 Int_t jlast = fYaxis.GetLast();
2655
2656 // Determine the size of the bin buffer(s) needed
2657 Double_t nentries = fEntries;
2658 Int_t nx = GetNbinsX();
2659 Int_t ny = GetNbinsY();
2660 Int_t bufSize = (nx + 2) * (ny + 2);
2661 auto* buf = new Double_t[bufSize];
2662 Double_t* ebuf = nullptr;
2663 if(fSumw2.fN != 0) {
2664 ebuf = new Double_t[bufSize];
2665 }
2666
2667 // Copy all the data to the temporary buffers
2668 Int_t i, j, bin;
2669 for(i = ifirst; i <= ilast; ++i) {
2670 for(j = jfirst; j <= jlast; ++j) {
2671 bin = GetBin(i, j);
2672 buf[bin] = GetBinContent(bin);
2673 if(ebuf != nullptr) {
2674 ebuf[bin] = GetBinError(bin);
2675 }
2676 }
2677 }
2678
2679 // Kernel tail sizes (kernel sizes must be odd for this to work!)
2680 Int_t x_push = (ksize_x - 1) / 2;
2681 Int_t y_push = (ksize_y - 1) / 2;
2682
2683 // main work loop
2684 for(i = ifirst; i <= ilast; ++i) {
2685 for(j = jfirst; j <= jlast; ++j) {
2686 Double_t content = 0.0;
2687 Double_t error = 0.0;
2688 Double_t norm = 0.0;
2689
2690 for(Int_t n = 0; n < ksize_x; ++n) {
2691 for(Int_t m = 0; m < ksize_y; ++m) {
2692 Int_t xb = i + (n - x_push);
2693 Int_t yb = j + (m - y_push);
2694 if((xb >= 1) && (xb <= nx) && (yb >= 1) && (yb <= ny)) {
2695 bin = GetBin(xb, yb);
2696 Double_t k = kernel[n * ksize_y + m];
2697 if(k != 0.0) {
2698 norm += k;
2699 content += k * buf[bin];
2700 if(ebuf != nullptr) {
2701 error += k * k * ebuf[bin] * ebuf[bin];
2702 }
2703 }
2704 }
2705 }
2706 }
2707
2708 if(norm != 0.0) {
2709 SetBinContent(i, j, content / norm);
2710 if(ebuf != nullptr) {
2711 error /= (norm * norm);
2712 SetBinError(i, j, sqrt(error));
2713 }
2714 }
2715 }
2716 }
2717 fEntries = nentries;
2718
2719 delete[] buf;
2720 delete[] ebuf;
2721}
