22public:
23 ArgParseItem(bool firstPass) : present_(false), fFirstPass(firstPass) {}
24 virtual ~ArgParseItem() = default;

26 virtual void parse_item(const std::vector<std::string>& arguments) = 0;
27 virtual int num_arguments() const = 0;
28 virtual std::string printable(int description_column = -1, int* chars_before_desc = nullptr) const = 0;

33 void parse(const std::string& name, const std::vector<std::string>& arguments, bool firstPass,
34 bool ignore_num_arguments = false)
35 {
36 if(firstPass != fFirstPass) {
37 return;
38 }
39 if(!ignore_num_arguments) {
40 if((num_arguments() == -1 && arguments.size() == 0) ||
41 (num_arguments() != -1 && arguments.size() != size_t(num_arguments()))) {
42 std::stringstream ss;
43 if(num_arguments() == -1) {
44 ss << R"(Flag ")" << name << R"(" expected at least one argument)";
45 } else {
46 ss << R"(Flag ")" << name << R"(" expected )" << num_arguments() << " argument(s) and received "
47 << arguments.size();
48 }
49 throw ParseError(ss.str());
50 }
51 }
52
53 present_ = true;
54 parse_item(arguments);
55 }
56
57private:
58 bool present_;
59 bool fFirstPass;

298/// Example usage:
299/// ```
300/// ArgParser parser;

324 void parse(int argc, char** argv, bool firstPass)
325 {
326 /// this version takes argc and argv, parses them, and sets only those
327 /// that have the matching firstPass flag set
328 /// this allows us to parse command line arguments in two stages, one
329 /// to get the normal options and file names (from which the run info
330 /// and analysis options are read), and a second stage where only the
331 /// run info and analysis option flags are parsed
332 bool double_dash_encountered = false;
333
334 int iarg = 1;
335 while(iarg < argc) {
336 std::string arg = argv[iarg++];
337
338 if(arg.at(0) != '-' || double_dash_encountered) {
339 handle_default_option(argc, argv, iarg, firstPass);
340 } else if(arg.substr(0, 2) == "--") {
341 handle_long_flag(argc, argv, iarg, firstPass);
342 } else {
343 handle_short_flag(argc, argv, iarg, firstPass);
344 }
345 }
346
347 for(auto& val : values) {
348 if(val->is_required() && !val->is_present()) {
349 std::stringstream ss;
350 ss << R"(Required argument ")" << val->flag_name() << R"(" is not present)";
351 throw ParseError(ss.str());
352 }
353 }
354 }

356 void parse_file(std::string& filename)
357 {
358 std::ifstream infile(filename);
359
360 std::string line;
361 while(std::getline(infile, line)) {
362 size_t colon = line.find(':');
363 bool has_colon = (colon != std::string::npos);
364
365 std::string flag;
366 std::string remainder;
367 if(has_colon) {
368 flag = line.substr(0, colon);
369 std::stringstream(flag) >> flag; // Strip out whitespace
370 if(flag.length() == 1) {
371 flag = "-" + flag;
372 } else {
373 flag = "--" + flag;
374 }
375 remainder = line.substr(colon + 1, line.length());
376 } else {
377 flag = "";
378 remainder = line;
379 }
380
381 std::vector<std::string> args;
382 std::stringstream ss(remainder);
383 std::string tmparg;
384 while(ss >> tmparg) {
385 args.push_back(tmparg);
386 }
387
388 if(has_colon) {
389 ArgParseItem& item = get_item(flag);
390 item.parse(flag, args, true, true); // parse "firstPass" items only
391 item.parse(flag, args, false, true); // parse "!firstPass" items only
392 } else {
393 for(auto& arg : args) {
394 ArgParseItem& item = get_item(arg);
395 item.parse(arg, std::vector<std::string>{arg}, true); // parse "firstPass" items only
396 item.parse(arg, std::vector<std::string>{arg}, false); // parse "!firstPass" items only
397 }
398 }
399 }

437 void handle_long_flag(int argc, char** argv, int& iarg, bool firstPass)
438 {
439 std::string arg = argv[iarg - 1];
440 std::vector<std::string> flag_args;
441 std::string flag;
442 size_t equals_index = arg.find('=');
443 if(equals_index == std::string::npos) {
444 // flag followed by list of flag_args
445 flag = arg;
446 } else {
447 // = inside flag
448 flag = arg.substr(0, equals_index);
449 }
450
451 ArgParseItem& item = get_item(flag);
452
453 if(equals_index == std::string::npos) {
454 flag_args = argument_list(argc, argv, iarg, item.num_arguments());
455 } else {
456 flag_args.push_back(arg.substr(equals_index + 1));
457 }
458 item.parse(flag, flag_args, firstPass);
459 }
460
461 void handle_short_flag(int argc, char** argv, int& iarg, bool firstPass)
462 {
463 std::string arg = argv[iarg - 1];
464 std::string flag = arg.substr(0, 2);
465 ArgParseItem& item = get_item(flag);
466 if(item.num_arguments() == 0) {
467 // Each character is a boolean flag
468 for(unsigned int ichar = 1; ichar < arg.length(); ichar++) {
469 std::string tmpflag = "-" + arg.substr(ichar, 1);
470 std::vector<std::string> flag_args;
471 get_item(tmpflag).parse(tmpflag, flag_args, firstPass);
472 }
473 } else {
474 if(arg.length() == 2) {
475 // Next arguments passed to the program get passed to the flag.
476 std::vector<std::string> flag_args = argument_list(argc, argv, iarg, item.num_arguments());
477 item.parse(flag, flag_args, firstPass);
478 } else {
479 // Everything past the first character is argument to the flag.
480 std::vector<std::string> flag_args{arg.substr(2)};
481 item.parse(flag, flag_args, firstPass);
482 }
483 }

495 //! Reads arguments into a list until finding one that begins with '-'
496 std::vector<std::string> argument_list(int argc, char** argv, int& iarg, int max_args)
497 {
498 std::vector<std::string> output;
499 bool read_extra = false;
500 while(iarg < argc && (max_args == -1 || output.size() < size_t(max_args))) {
501 std::string next_arg = argv[iarg++];
502 if(next_arg.length() && next_arg.at(0) == '-') {
503 read_extra = true;
504 break;
505 } else {
506 output.push_back(next_arg);
507 }
508 }
509 if(read_extra) {
510 iarg--;
511 }
512 return output;
513 }
