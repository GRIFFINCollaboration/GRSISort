79bool TEventBuildingLoop::Iteration()
80{
81 // Pull something off of the input queue.
82 std::shared_ptr<const TFragment> input_frag = nullptr;
83 fInputSize = fInputQueue->Pop(input_frag, 0);
84 if(fInputSize < 0) {
85 fInputSize = 0;
86 }
87
88 if(input_frag != nullptr) {
89 ++fItemsPopped;
90 if(!fSkipInputSort) {
91 fOrdered.insert(input_frag);
92 if(fOrdered.size() < fSortingDepth) {
93 // Got a new event, but we want to have more to sort
94 return true;
95 }
96 // Got a new event, and we have enough to sort.
97 }
98 } else {
99 if(!fInputQueue->IsFinished()) {
100 // If the parent is live, wait for it
101 std::this_thread::sleep_for(std::chrono::milliseconds(1000));
102 return true;
103 }
104 if(fOrdered.empty()) {
105 // Parent is dead, and we have passed on all events
106 // check if last event needs to be pushed
107 if(!fNextEvent.empty()) {
108 fOutputQueue->Push(fNextEvent);
109 }
110 fOutputQueue->SetFinished();
111 return false;
112 }
113 // Parent is dead, but we still have items.
114 // Continue through the function to process them.
115 }
116
117 // We have data, and we want to add it to the next fragment;
118 std::shared_ptr<const TFragment> next_fragment;
119 if(!fSkipInputSort) {
120 next_fragment = *fOrdered.begin();
121 fOrdered.erase(fOrdered.begin());
122 } else {
123 next_fragment = input_frag;
124 }
125
126 if(CheckBuildCondition(next_fragment)) {
127 fNextEvent.push_back(next_fragment);
128 }
129
130 return true;
131}

133bool TEventBuildingLoop::CheckBuildCondition(const std::shared_ptr<const TFragment>& frag)
134{
135 switch(fBuildMode) {
136 case EBuildMode::kTime: return CheckTimeCondition(frag); break;
137 case EBuildMode::kTimestamp: return CheckTimestampCondition(frag); break;
138 case EBuildMode::kTriggerId: return CheckTriggerIdCondition(frag); break;
139 case EBuildMode::kSkip:
140 // always push the current "event" (single fragment) on and clear it
141 fOutputQueue->Push(fNextEvent);
142 fNextEvent.clear();
143 return true;
144 break;
145 default: return false;
146 }
147 return false; // we should never reach this statement!
148}
