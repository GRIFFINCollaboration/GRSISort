121Short_t TLaBr::GetSuppressedMultiplicity(const TBgo* bgo)
122{
123 /// Automatically builds the suppressed hits using the fSuppressionCriterion and returns the number of suppressed hits
124 if(fHits.empty()) {
125 return 0;
126 }
127 // if the suppressed has been reset, clear the suppressed hits
128 if(!IsSuppressed()) {
129 for(auto& hit : fSuppressedHits) {

132 fSuppressedHits.clear();
133 }
134 if(fSuppressedHits.empty()) {
135 CreateSuppressed(bgo, fHits, fSuppressedHits);
136 SetSuppressed(true);
