7TGriffinAngles::TGriffinAngles(double distance, bool folding, bool grouping, bool addback)
8 : fDistance(distance), fFolding(folding), fGrouping(grouping), fAddback(addback)
9{
10 SetName("GriffinAngles");
11 // get user settings for excluded detectors/crystals
12 if(TGRSIOptions::Get() != nullptr) {
13 auto settings = TGRSIOptions::Get()->UserSettings();
14 if(settings != nullptr) {
15 // try quietly to get the vectors of excluded crystals and detectors, catching (and disregarding) any exceptions
16 try {
17 fExcludedCrystals = settings->GetIntVector("ExcludedCrystal", true);
18 } catch(std::out_of_range&) {}
19 try {
20 fExcludedDetectors = settings->GetIntVector("ExcludedDetector", true);
21 } catch(std::out_of_range&) {}
22 } else {
23 std::cout<<"Failed to find user settings in TGRSIOptions, can't get user settings for excluded detectors/crystals"<<std::endl;
24 }
25 } else {
26 std::cout<<"Failed to find TGRSIOptions, can't get user settings for excluded detectors/crystals"<<std::endl;
27 }
28
29 // loop over all possible detector/crystal combinations
30 for(int firstDet = 1; firstDet <= 16; ++firstDet) {
31 if(ExcludeDetector(firstDet)) continue;
32 for(int firstCry = 0; firstCry < 4; ++firstCry) {
33 if(ExcludeCrystal(firstDet, firstCry)) continue;
34 for(int secondDet = 1; secondDet <= 16; ++secondDet) {
35 if(ExcludeDetector(secondDet)) continue;
36 for(int secondCry = 0; secondCry < 4; ++secondCry) {
37 if(ExcludeCrystal(secondDet, secondCry)) continue;
38 // exclude hits in the same crystal or, if addback is enabled, in the same detector
39 if(firstDet == secondDet && (firstCry == secondCry || fAddback)) continue;
40 double angle = TGriffin::GetPosition(firstDet, firstCry, fDistance).Angle(TGriffin::GetPosition(secondDet, secondCry, fDistance)) *180./TMath::Pi();
41 // if folding is enable we fold the distribution at 90 degree and only use angles between 0 and 90 degree
42 if(fFolding && angle > 90.) {
43 angle = 180. - angle;
44 }
45 // if the lower bound and the upper bound are the same we have a new angle
46 if(fAngles.lower_bound(angle-fRounding) == fAngles.upper_bound(angle+fRounding)) {
47 fAngles.insert(angle);
48 }
49 // this should always work, either this is a new angle, in which case it gets initialized to zero and then incremented to one,
50 // or we increment the existing counter
51 // the key is integer, so by dividing by rounding and then casting to integer we can avoid duplicates close to each other
52 // factor 2 to include that the "normal" rounding is +- fRounding
53 fAngleCount[std::round(angle/fRounding)]++;
54 } // second crystal loop
55 } //second detector loop
56 } // first crystal loop
57 } // first detector loop
58
59 // create map of indices before we group so that we have an index for each (folded) angle
60 for(auto it = fAngles.begin(); it != fAngles.end(); ++it) {
61 fAngleMap.insert(std::make_pair(*it, std::distance(fAngles.begin(), it)));
62 }
63
64 if(fGrouping) {
65 // If grouping is enable we group angles that are close to each other together.
66 // Which angles are grouped is somewhat arbitrary, this grouping was chosen to get similar numbers of detector combinations and thus statistics for each angle group.
67 // Due to the way lower_bound works, we use the highest angle of each group as the angle of that group.
68 // This is just for the purpose of this algorithm, when plotting the correct average angle of the group should be used!
69 std::set<double> groupedAngles;
70 // for the angle map we need to update the indices: 3->2, 4->3, 5->3, 6->4, 7->4, 8->5, 9->5, 10->5, 11->6, 12->6, 13->6 and so on
71 for(auto it = fAngles.begin(); it != fAngles.end(); ++it) {
72 auto i = std::distance(fAngles.begin(), it);
73 switch(i) {
74 case 0:
75 case 1:
76 // first and second angle are not grouped
77 groupedAngles.insert(*it);
78 break;
79 case 2:
80 case 4:
81 case 6:
82 // three groups of two angles each
83 // 2->2, 4->3, 6->4
84 fAngleMap[*it] -= (i-2)/2;
85 ++it;
86 // 3->2, 5->3, 7->4
87 fAngleMap[*it] -= (i-2)/2+1;
88 groupedAngles.insert(*it);
89 break;
90 default:
91 // all others are groups of three
92 // 8->5, 11->6, 14->7, ...
93 fAngleMap[*it] -= (i-5)/3*2+1;
94 ++it;
95 // 9->5, 12->6, 15->7, ...
96 fAngleMap[*it] -= (i-5)/3*2+2;
97 ++it;
98 // 10->5, 13->6, 16->7, ...
99 fAngleMap[*it] -= (i-5)/3*2+3;
100 groupedAngles.insert(*it);
101 break;
102 }
103 }
104 fAngles = groupedAngles;
105 }

108int TGriffinAngles::Index(double angle)
109{
110 auto matches = [angle, this](std::pair<double, int> val) { return std::abs(val.first-angle) < fRounding; };
111 auto it = std::find_if(fAngleMap.begin(), fAngleMap.end(), matches);
112
113 if(it == fAngleMap.end()) {
114 std::cout<<"Failed to find angle "<<angle<<" in map!"<<std::endl;
115 return -1;
116 }
117 if(it->second >= static_cast<int>(fAngles.size())) {
118 std::cout<<"Found index "<<it->second<<" for angle "<<angle<<" which is outside range ("<<fAngles.size()<<")"<<std::endl;
119 return -1;
120 }
121 return it->second;
122}
