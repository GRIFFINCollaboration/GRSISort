92 def write(self, text, is_editable=False):
93 """
94 add text to the text widget
95 """
96 # first split the text at color codes, stripping stuff like the <ESC>
97 # and \[ characters and keeping only the inner "0;23"-like codes
98 segments = AnsiColorText.color_pat.split(text)
99 if segments:
100 for text in segments:
101 # a segment can be regular text, or it can be a color pattern
102 if AnsiColorText.inner_color_pat.match(text):
103 # if it's a color pattern, check if we already have
104 # registered a tag for it
105 if text not in self.known_tags:
106 # if tag not yet registered,
107 # extract the foreground and background color
108 # and ignore the other things
109 parts = text.split(";")
110 for part in parts:
111 if part in AnsiColorText.foreground_colors[self.bright]:
112 self.foregroundcolor = AnsiColorText.foreground_colors[self.bright][part]
113 elif part in AnsiColorText.background_colors[self.bright]:
114 self.backgroundcolorbackgroundcolor = AnsiColorText.background_colors[self.bright][part]
115 else:
116 for ch in part:
117 if ch == '0' :
118 # reset all attributes
119 self.reset_to_default_attribs()
120 if ch == '1' :
121 # define bright colors
122 self.bright = 'bright'
123 if ch == '2' :
124 # define dim colors
125 self.bright = 'dim'
126
127 self.register_tag(text,
128 foreground=self.foregroundcolor,
129 background=self.backgroundcolorbackgroundcolor)
130 # remember that we switched to this tag
131 self.tag = text
132 elif text == '':
133 # reset tag to black
134 self.tag = '30' # black
135 else:
136 # no color pattern, insert text with the currently selected
137 # tag
138 self.insert(END,text,self.tag)
