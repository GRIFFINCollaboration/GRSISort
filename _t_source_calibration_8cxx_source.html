493 auto peakPos = std::get<0>(iter->first);
494 auto peakPosErr = std::get<1>(iter->first);
495 auto peakArea = std::get<2>(iter->first);
496 auto peakAreaErr = std::get<3>(iter->first);
497 auto energy = std::get<0>(iter->second);
498 auto energyErr = std::get<1>(iter->second);
499 auto intensity = std::get<2>(iter->second);
500 auto intensityErr = std::get<3>(iter->second);
501 // drop this peak if the uncertainties in area or position are too large
502 if(peakPosErr > 0.1*peakPos || peakAreaErr > peakArea) {
503 if(fVerboseLevel > 1) std::cout<<"Dropping peak with position "<<peakPos<<" +- "<<peakPosErr<<", area "<<peakArea<<" +- "<<peakAreaErr<<", energy "<<energy<<", intensity "<<intensity<<std::endl;
504 map.erase(iter++);
505 } else {
506 fData->SetPoint(i, peakPos, energy);
507 fData->SetPointError(i, peakPosErr, energyErr);
508 // we use the source energy for the x-values, and the ratio of peak area and intensity for y
509 fEfficiency->SetPoint(i, energy, peakArea/intensity);
510 fEfficiency->SetPointError(i, energyErr, peakArea/intensity*TMath::Sqrt(TMath::Power(peakAreaErr/peakArea,2)+TMath::Power(intensityErr/intensity,2)));
511 if(fVerboseLevel > 2) std::cout<<"Using peak with position "<<peakPos<<" +- "<<peakPosErr<<", area "<<peakArea<<" +- "<<peakAreaErr<<", energy "<<energy<<", intensity "<<intensity<<std::endl;
512 ++iter;
513 ++i;
514 }
515 }
516 // if we dropped a peak, we need to resize the graph, if we haven't dropped any this doesn't do anything
517 fData->Set(map.size());
518 fEfficiency->Set(map.size());
519 // remove poly markers that weren't used for the fit
520 TList* functions = fProjection->GetListOfFunctions();
521 TPolyMarker* pm = static_cast<TPolyMarker*>(functions->FindObject("TPolyMarker"));
522 double* oldX = pm->GetX();
523 double* oldY = pm->GetY();
524 int size = pm->GetN();
525 std::vector<double> newX;
526 std::vector<double> newY;
527 for(i = 0; i < size; ++i) {
528 for(auto point : map) {
529 if(TMath::Abs(oldX[i] - std::get<0>(point.first)) < fSigma) {
530 newX.push_back(oldX[i]);
531 newY.push_back(oldY[i]);
532 break;
533 }
534 }
535 }
536 pm->SetPolyMarker(newX.size(), newX.data(), newY.data());
537 // remove fit functions for unused peaks
538 TIter iter(functions);
539 TObject* item;
540 while((item = iter.Next()) != nullptr) {
541 if(item->IsA() == TF1::Class() || item->IsA() == GPeak::Class()) { // if the item is a TF1 or GPeak we see if we can find the centroid in the map of used peaks
542 double centroid = 0.;
543 if(item->IsA() == TF1::Class()) centroid = static_cast<TF1*>(item)->GetParameter(1);
544 else centroid = static_cast<GPeak*>(item)->Centroid();
545 bool found = false;
546 for(auto point : map) {
547 if(TMath::Abs(centroid - std::get<0>(point.first)) < fSigma) {
548 found = true;
549 break;
550 }
551 }
552 // remove the TF1 if it wasn't found in the map
553 if(!found) {
554 functions->Remove(item);
555 }
556 }
557 }
558}

701 }
702 // we looped over all bins without finding any issues, so we can just copy the first matrix
703 fActualChannelId[i] = fActualChannelId[0];
704 fData[i].resize(fNofBins, nullptr);
705 fEfficiency[i].resize(fNofBins, nullptr);
706 }

1035void TSourceCalibration::Navigate(Int_t id)
1036{
1037 // we get the current source tab id and use it to get the channel tab from the right source tab
1038 // since the current id only refers to the position within the open tabs we need to keep track of the actual id it relates to
1039 // for this we created a vector with all ids at the beginning and now remove the position correspoding to the current id when we remove the tab
1040 int currentSourceId = fTab->GetCurrent();
1041 int actualSourceId = fActualSourceId[currentSourceId];
1042 auto currentTab = fSourceTab[actualSourceId]->ChannelTab();
1043 int currentChannelId = currentTab->GetCurrent();
1044 int nofTabs = currentTab->GetNumberOfTabs();
1045 if(fVerboseLevel > 1) std::cout<<__PRETTY_FUNCTION__<<": id "<<id<<", source tab id "<<currentSourceId<<", actual source tab id "<<actualSourceId<<", channel tab id "<<currentTab->GetCurrent()<<", # of tabs "<<nofTabs<<std::endl;
1046 switch(id) {
1047 case 1: // previous
1048 currentTab->SetTab(currentChannelId-1);
1049 SelectedTab(currentChannelId-1);
1050 break;
1051 case 2: // find peaks
1052 FindPeaks();
1053 break;
1054 case 3: // find peaks fast
1055 FindPeaksFast();
1056 break;
1057 case 4: // calibrate
1058 Calibrate();
1059 break;
1060 case 5: // discard
1061 // select the next (or if we are on the last tab, the previous) tab
1062 if(currentChannelId < nofTabs - 1) {
1063 currentTab->SetTab(currentChannelId+1);
1064 } else {
1065 currentTab->SetTab(currentChannelId-1);
1066 }
1067 // remove the original active tab
1068 currentTab->RemoveTab(currentChannelId);
1069 break;
1070 case 6: // accept
1071 AcceptChannel(currentChannelId);
1072 break;
1073 case 7: // accept all (no argument = -1 = all)
1074 AcceptChannel();
1075 break;
1076 case 8: // next
1077 currentTab->SetTab(currentChannelId+1);
1078 SelectedTab(currentChannelId+1);
1079 break;
1080 default:
1081 break;
1082 }
