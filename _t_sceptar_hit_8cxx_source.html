42 if(false) {
43 std::vector<Short_t> x;
44 // Need to reorder waveform data for S1507 data from December 2014
45 // All pairs of samples are swapped.
46 // The first two samples are also delayed by 8.
47 // We choose to throw out the first 2 samples (junk) and the last 6 samples (convience)

51

54 x[i] = x[i + 9];
55 x[i + 1] = x[i + 8];
56 temp = x[i + 2];
57 x[i + 2] = x[i + 3];
58 x[i + 3] = temp;
59 temp = x[i + 4];
60 x[i + 4] = x[i + 5];
61 x[i + 5] = temp;
62 temp = x[i + 6];
63 x[i + 6] = x[i + 7];
64 x[i + 7] = temp;
65 }
66 x.resize(length - 8);
67 }

70 frag.CopyWave(*this);

137bool TSceptarHit::AnalyzeWaveform()
138{
139 // Calculates the cfd time from the waveform
140 bool error = false;
141 if(fWaveform.empty()) {
142 return false; // Error!
143 }
144
145 std::vector<Int_t> baselineCorrections(8, 0);
146 std::vector<Short_t> smoothedWaveform;
147
148 // all timing algorithms use interpolation with this many steps between two samples (all times are stored as
149 // integers)
150 unsigned int interpolationSteps = 256;
151 int delay = 8;
152 double attenuation = 24. / 64.;
153 int halfsmoothingwindow = 0; // 2*halfsmoothingwindow + 1 = number of samples in moving window.
154
155 // baseline algorithm: correct each adc with average of first two samples in that adc
156 for(size_t i = 0; i < 8 && i < fWaveform.size(); ++i) {
157 baselineCorrections[i] = fWaveform[i];
158 }
159 for(size_t i = 8; i < 16 && i < fWaveform.size(); ++i) {
160 baselineCorrections[i - 8] =
161 ((baselineCorrections[i - 8] + fWaveform[i]) + ((baselineCorrections[i - 8] + fWaveform[i]) > 0 ? 1 : -1)) >>
162 1;
163 }
164 for(size_t i = 0; i < fWaveform.size(); ++i) {
165 fWaveform[i] -= baselineCorrections[i % 8];
166 }
167
168 SetCfd(CalculateCfd(attenuation, delay, halfsmoothingwindow, interpolationSteps));
169
170 return !error;
171}

182Int_t TSceptarHit::CalculateCfdAndMonitor(double attenuation, unsigned int delay, int halfsmoothingwindow,
183 unsigned int interpolationSteps, std::vector<Short_t>& monitor)
184{
185 // Used when calculating the CFD from the waveform
186
187 Short_t monitormax = 0;
188
189 bool armed = false;
190
191 Int_t cfd = 0;
192
193 std::vector<Short_t> smoothedWaveform;
194
195 if(fWaveform.empty()) {
196 return INT_MAX; // Error!
197 }
198
199 if(static_cast<unsigned int>(fWaveform.size()) > delay + 1) {
200
201 if(halfsmoothingwindow > 0) {
202 smoothedWaveform = TSceptarHit::CalculateSmoothedWaveform(halfsmoothingwindow);
203 } else {
204 smoothedWaveform = fWaveform;
205 }
206
207 monitor.resize(smoothedWaveform.size() - delay);
208 monitor[0] = attenuation * smoothedWaveform[delay] - smoothedWaveform[0];
209 if(monitor[0] > monitormax) {
210 armed = true;
211 monitormax = monitor[0];
212 }
213
214 for(unsigned int i = delay + 1; i < smoothedWaveform.size(); ++i) {
215 monitor[i - delay] = attenuation * smoothedWaveform[i] - smoothedWaveform[i - delay];
216 if(monitor[i - delay] > monitormax) {
217 armed = true;
218 monitormax = monitor[i - delay];
219 } else {
220 if(armed && monitor[i - delay] < 0) {
221 armed = false;
222 if(monitor[i - delay - 1] - monitor[i - delay] != 0) {
223 // Linear interpolation.
224 cfd = (i - delay - 1) * interpolationSteps +
225 (monitor[i - delay - 1] * interpolationSteps) / (monitor[i - delay - 1] - monitor[i - delay]);
226 } else {
227 // Should be impossible, since monitor[i-delay-1] => 0 and monitor[i-delay] > 0
228 cfd = 0;
229 }
230 }
231 }
232 }
233 } else {
234 monitor.resize(0);
235 }
236
237 return cfd;
238}

240std::vector<Short_t> TSceptarHit::CalculateSmoothedWaveform(unsigned int halfsmoothingwindow)
241{
242 // Used when calculating the CFD from the waveform
243
244 if(fWaveform.empty()) {
245 return std::vector<Short_t>(); // Error!
246 }
247
248 std::vector<Short_t> smoothedWaveform(std::max(static_cast<size_t>(0), fWaveform.size() - 2 * halfsmoothingwindow),
249 0);
250
251 for(size_t i = halfsmoothingwindow; i < fWaveform.size() - halfsmoothingwindow; ++i) {
252 for(int j = -static_cast<int>(halfsmoothingwindow); j <= static_cast<int>(halfsmoothingwindow); ++j) {
253 smoothedWaveform[i - halfsmoothingwindow] += fWaveform[i + j];
254 }
255 }
256
257 return smoothedWaveform;
258}
