41void TReaction::InitReaction()
42{
43
44 // An initializing function that sets the energies and momenta of the beam and target in the lab and CM frame,
45 // as well as a few basic calculations.
46
47 fTLab[0] = fTBeam; // target kinetic energy is ebeam
48 fELab[0] = fTLab[0] + fM[0]; // total E of beam in lab
49 fPLab[0] = sqrt(pow(fTLab[0], 2) + 2 * fTLab[0] * fM[0]); // momentum of beam in lab
50 fVLab[0] = fPLab[0] / fELab[0]; // velocity of beam in lab
51 fGLab[0] = 1 / sqrt(1 - pow(fVLab[0], 2)); // gamma factor of beam
52
53 fTLab[1] = 0; // target kinetic energy is always 0
54 fELab[1] = fM[1]; // total E of target in lab
55 fPLab[1] = 0; // momentum of target in lab
56 fVLab[1] = 0; // velocity of target in lab
57 fGLab[1] = 1; // gamma factor of target
58
59 fS = pow(fM[0], 2) + pow(fM[1], 2) + 2 * fELab[0] * fM[1]; // fELab[1] = fM[1]
60 fInvariantMass = sqrt(fS); // âˆšs
61
62 // CM motion
63 fCmE = fInvariantMass;
64 fCmTi = fCmE - fM[0] - fM[1];
65 fCmTf = fCmTi + fQVal;
66 fCmV = fPLab[0] / (fELab[0] + fM[1]);
67 fCmP = fCmV * fCmE;
68 fCmG = 1 / sqrt(1 - pow(fCmV, 2));
69
70 // take care of particles in CM frame using the reaction excitation energy
71 SetCmFrame(fExc);
72
73 // options to make graphs draw nicely
74 gStyle->SetTitleYOffset(1.5);
75 gStyle->SetTitleXOffset(1.2);
76 gStyle->SetDrawOption("AC");
77}

87void TReaction::SetCmFrame(double exc)
88{
89
90 // particles in CM frame
91 fPCm[0] = sqrt((fS - pow(fM[0] - fM[1], 2)) * (fS - pow(fM[0] + fM[1], 2))) / (2 * sqrt(fS));
92 fPCm[1] = fPCm[0];
93 fPCm[2] = sqrt((fS - pow(exc + fM[2] - fM[3], 2)) * (fS - pow(exc + fM[2] + fM[3], 2))) / (2 * sqrt(fS));
94 fPCm[3] = fPCm[2];
95
96 for(int i = 0; i < 4; i++) {
97 fECm[i] = sqrt(pow(fM[i], 2) + pow(fPCm[i], 2));
98 fTCm[i] = fECm[i] - fM[i];
99 fVCm[i] = fPCm[i] / fECm[i];
100 fGCm[i] = 1 / sqrt(1 - pow(fVCm[i], 2));
101
102 // max lab frame theta
103 if(i < 2) {
104 fThetaMax[i] = 0;
105 } else {
106 double val = fPCm[i] / (fM[i] * fCmV * fCmG);
107 if(val < 1) {
108 fThetaMax[i] = asin(val);
109 } else if(val < 1.001) { // catches elastic channels with small numerical rounding errors
110 fThetaMax[i] = PI / 2;
111 } else {
112 fThetaMax[i] = PI;
113 }
114 }
115 }
116}

150double TReaction::GetPLabFromThetaCm(double theta_cm, int part)
151{
152 if(part == 0 || part == 1) {
153 return fTLab[part];
154 }
155
156 double Pz = fCmG * (fPCm[part] * cos(theta_cm) - fCmV * fECm[part]);
157 double Pperp = fPCm[part] * sin(theta_cm);
158 return sqrt(pow(Pperp, 2) + pow(Pz, 2));

171double TReaction::GetExcEnergy(double ekin, double theta_lab, int)
172{
173 if(ekin == 0.00 && theta_lab == 0.00) {
174 return fExc;
175 }
176
177 double val1 = pow(fM[0] + fM[1], 2) + pow(fM[2], 2) + 2 * fM[1] * fTLab[0];
178 double val2 = 2 * fCmG * sqrt(pow(fM[0] + fM[1], 2) + 2 * fM[1] * fTLab[0]);
179 double val3 = fM[2] + ekin - fCmV * sqrt(pow(ekin, 2) + 2 * fM[2] * ekin) * TMath::Cos(theta_lab);
180
181 return sqrt(val1 - val2 * val3) - fM[3];
182}

235// Conversion from LAB frame to CM frame
236double TReaction::ConvertThetaLabToCm(double theta_lab, int part)
237{
238
239 if(theta_lab > fThetaMax[part]) {
240 theta_lab = fThetaMax[part];
241 }
242
243 // Uses the particle velocity in the CM frame, which makes it more complex
244 double gtan2 = pow(tan(theta_lab) * fCmG, 2);
245 double x = fCmV / fVCm[part];
246 double expr = sqrt(1 + gtan2 * (1 - pow(x, 2)));
247 double theta_cm;
248
249 // deals with double valued thetas in lab frame
250 if(tan(theta_lab) >= 0) {
251 theta_cm = acos((-x * gtan2 + expr) / (1 + gtan2));
252 } else {
253 theta_cm = acos((-x * gtan2 - expr) / (1 + gtan2));
254 }
255
256 if(fInverse) {
257 theta_cm = PI - theta_cm;
258 }
259
260 if(part == 3) {
261 theta_cm = -theta_cm;
262 }
263
264 return theta_cm;
265}

281// Conversion from CM frame to LAB frame
282double TReaction::ConvertThetaCmToLab(double theta_cm, int part)
283{
284
285 if(fInverse) {
286 theta_cm = PI - theta_cm;
287 }
288
289 double theta_lab = TMath::ATan2(sin(theta_cm), fCmG * (cos(theta_cm) + fCmV / fVCm[part]));
290
291 if(theta_lab > fThetaMax[part]) {
292 return fThetaMax[part];
293 }
294 return theta_lab;
295}

297double TReaction::ConvertOmegaCmToLab(double theta_cm, int part)
298{
299
300 // the way to test this function is to use the known 4*cos(theta_lab) for elastics
301 double x = fCmV / fVCm[part];
302 if(fInverse) {
303 theta_cm = PI - theta_cm;
304 }
305 double val1 = pow(pow(sin(theta_cm), 2) + pow(fCmG * (x + cos(theta_cm)), 2), 1.5);
306 double val2 = (fCmG * (1 + x * cos(theta_cm)));
307
308 return val1 / val2;
309}
