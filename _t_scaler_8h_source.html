129 static double GetLastScaler(TTree* tree = nullptr, UInt_t address = 0xffff, size_t nominator = 2, size_t denominator = 1)
130 {
131 /// This function returns the ratio of the two scalers nominator and denominator for the last entry with a matching address for a given tree.
132 /// If no tree is provided it tries to get the "RateScaler" tree from the current directory.
133 if(tree == nullptr) {
134 tree = static_cast<TTree*>(gDirectory->Get("RateScaler"));
135 if(tree == nullptr) {
136 std::cerr<<__PRETTY_FUNCTION__<<": no tree provided and failed to find \"RateScaler\" in "<<gDirectory->GetName()<<std::endl;
137 return -1.;
138 }
139 }
140 TScalerData* scalerData = nullptr;
141 tree->SetBranchAddress("ScalerData", &scalerData);
142
143 for(Long64_t entry = tree->GetEntries()-1; entry >= 0; --entry) {
144 tree->GetEntry(entry);
145 if(scalerData->GetAddress() != address) continue;
146 auto scalers = scalerData->GetScaler();
147 if(nominator >= scalers.size() || denominator >= scalers.size()) {
148 std::cerr<<__PRETTY_FUNCTION__<<": trying to get nominator "<<nominator<<" and denominator "<<denominator<<" from vector of size "<<scalers.size()<<std::endl;
149 return -1.;
150 }
151 return ((double)scalers[nominator])/scalers[denominator];
152 }
153 std::cerr<<__PRETTY_FUNCTION__<<": failed to find any entry for address "<<hex(address, 4)<<std::endl;
154 return -1.;
155 }
