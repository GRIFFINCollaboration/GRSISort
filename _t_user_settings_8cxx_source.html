14 bool TUserSettings::ReadSettings(std::string settingsFile)
15{
16 /// Read user settings from text file.
17 /// The file is expected to have the format
18 /// parameter name: value
19 /// where parameter name is a string w/o whitespace and value is the value of the parameter
20 /// the type of the value (bool, int, double, std::string, or vectors of these types) will be determined automatically
21 /// vectors are comma separated values of one type
22 /// the default type will be std::string
23 /// any lines starting with '#' or '//' or without a colon will be ignored
24 SetName("UserSettings");
25 fSettingsFiles.push_back(settingsFile);
26
27 std::ifstream settings(settingsFile);
28
29 if(!settings.is_open()) {
30 std::cerr << "Failed to open user settings file '" << settingsFile << "'!" << std::endl;
31 return false;
32 }
33
34 std::string line;
35 while(std::getline(settings, line)) {
36 // skip lines with too few character, starting with '#' or "//", or w/o colon
37 if(line.length() < 3) continue; // need at least three characters
38 if(line[0] == '#') continue;
39 if(line[0] == '/' && line[1] == '/') continue;
40 auto colon = line.find(':');
41 if(colon == std::string::npos) continue;
42 // split line at colon
43 auto name = line.substr(0, colon);
44 auto value = line.substr(colon + 1);
45 // remove leading and trailing whitespace
46 trim(value);
47
48 // check if this is a comma separated list of values
49 if(value.find(',') != std::string::npos) {
50 std::stringstream valueStream(value);
51 while(std::getline(valueStream, line, ',')) {
52 trim(line);
53 ParseValue(name, line, true);
54 }
55 } else {
56 ParseValue(name, value, false);
57 }
58 }
59
60 return true;
61}

63void TUserSettings::ParseValue(const std::string& name, const std::string& value, bool vector)
64{
65 // try and parse as integer, if pos is equal to the length of the string we were successful:
66 // add it to the map and go to the next line
67 size_t pos;
68 try {
69 auto intVal = std::stoi(value, &pos);
70 if(pos == value.length()) {
71 if(!vector) {
72 fInt[name] = intVal;
73 } else {
74 fIntVector[name].push_back(intVal);
75 }
76 return;
77 }
78 } catch(std::invalid_argument& e) {
79 // do nothing, we failed to parse the input as integer, just go on to the next type
80 }
81
82 // try and parse as double, if pos is equal to the length of the string we were successful:
83 // add it to the map and go to the next line
84 try {
85 auto doubleVal = std::stod(value, &pos);
86 if(pos == value.length()) {
87 if(!vector) {
88 fDouble[name] = doubleVal;
89 } else {
90 fDoubleVector[name].push_back(doubleVal);
91 }
92 return;
93 }
94 } catch(std::invalid_argument& e) {
95 // do nothing, we failed to parse the input as double, just go on to the next type
96 }
97
98 // try and parse as bool: convert a copy to lower case and use stringstream with std::boolalpha
99 std::string copy = value;
100 std::transform(copy.begin(), copy.end(), copy.begin(), ::tolower);
101 bool boolVal;
102 std::stringstream str(copy);
103 str >> std::boolalpha >> boolVal;
104 if(str.good()) {
105 if(!vector) {
106 fBool[name] = boolVal;
107 } else {
108 fBoolVector[name].push_back(boolVal);
109 }
110 return;
111 }
112
113 // trying to parse as integer, double, or bool failed, so we assume it's a string and just add it to the map
114 if(!vector) {
115 fString[name] = value;
116 } else {
117 fStringVector[name].push_back(value);
118 }
119}
