24Bool_t TRunInfo::ReadInfoFromFile(TFile* tempf)
25{
26 TDirectory* savdir = gDirectory;
27 if(tempf != nullptr) {
28 tempf->cd();
29 }
30
31 if((gDirectory->GetFile()) == nullptr) {
32 std::cout<<"File does not exist"<<std::endl;
33 savdir->cd();
34 return false;
35 }
36
37 tempf = gDirectory->GetFile();
38
39 // try and read the run info directly
40 if(tempf->FindObjectAny("TRunInfo") != nullptr) {
41 Get()->fDetectorInformation = static_cast<TDetectorInformation*>(tempf->Get("TRunInfo"));
42 savdir->cd();
43 return true;
44 }
45
46 // if the above failed we try all keys to find a run info
47 TList* list = tempf->GetListOfKeys();
48 TIter iter(list);
49 TKey* key;
50 while((key = static_cast<TKey*>(iter.Next())) != nullptr) {
51 if(strcmp(key->GetClassName(), "TRunInfo") != 0) {
52 continue;
53 }
54
55 Set(static_cast<TRunInfo*>(key->ReadObj()));
56 Get()->fDetectorInformation = nullptr;// just to be safe
57 // see if we can find detector information
58 while((key = static_cast<TKey*>(iter.Next())) != nullptr) {
59 // for some reason the classic way of using TClass::InheritsFrom fails in grsiproof
60 // so instead we just try to dynamically cast every key into TDetectorInformation
61 Get()->fDetectorInformation = dynamic_cast<TDetectorInformation*>(key->ReadObj());
62 if(Get()->fDetectorInformation != nullptr) break;
63 }
64 savdir->cd();
65 return true;
66 }
67 savdir->cd();
68
69 return false;
70}

212Bool_t TRunInfo::ParseInputData(const char* inputdata, Option_t* opt)
213{
214 // A helper function to parse the run info file.
215
216 std::istringstream infile(inputdata);
217 std::string line;
218 int linenumber = 0;
219
220 // Parse the info file.
221 while(!std::getline(infile, line).fail() ) {
222 linenumber++;
223 trim(line);
224 size_t comment = line.find("//");
225 if(comment != std::string::npos) {
226 line = line.substr(0, comment);
227 }
228 if(line.length() == 0u) {
229 continue;
230 }
231
232 size_t ntype = line.find(':');
233 if(ntype == std::string::npos) {
234 continue; // no seperator, not useful.
235 }
236
237 std::string type = line.substr(0, ntype);
238 line = line.substr(ntype + 1, line.length());
239 trim(line);
240 int j = 0;
241 while(type[j] != 0) {
242 char c = *(type.c_str() + j);
243 c = toupper(c);
244 type[j++] = c;
245 }
246 if(type.compare("CAL") == 0 || type.compare("CALFILE") == 0) {
247 // TODO Make this work again, using priorities
248 // TGRSIOptions::AddInputCalFile(line);
249 } else if(type.compare("MID") == 0 || type.compare("MIDAS") == 0 || type.compare("MIDASFILE") == 0) {
250 // TODO Make this work again, using priorities
251 // TGRSIOptions::AddInputMidasFile(line);
252 } else if(type.compare("ARRAYPOS") == 0 || type.compare("HPGEPOS") == 0) {
253 std::istringstream ss(line);
254 double temp_double;
255 ss >> temp_double;
256 Get()->SetHPGeArrayPosition(temp_double);
257 } else if(type.compare("BADCYCLE") == 0) {
258 std::istringstream ss(line);
259 int tmp_int;
260 while(!(ss >> tmp_int).fail() ) {
261 Get()->AddBadCycle(tmp_int);
262 }
263 }
264 }
265
266 if(strcmp(opt, "q") != 0) {
267 std::cout<<"parsed "<<linenumber<<" lines."<<std::endl;
268 std::cout<<DBLUE "\tArray Position (mm) = "<<DRED<<TRunInfo::HPGeArrayPosition()<<RESET_COLOR<<std::endl;
269 }
270 return true;
271}

405void TRunInfo::Add(TRunInfo* runinfo, bool verbose)
406{
407 // add new run to list of runs (and check if the current run needs to be added)
408 if(fRunList.empty()) {
409 std::pair<int, int> currentPair = std::make_pair(fRunNumber, fSubRunNumber);
410 fRunList.push_back(currentPair);
411 }
412 std::pair<int, int> newPair = std::make_pair(runinfo->fRunNumber, runinfo->fSubRunNumber);
413 // check for dual entries
414 if(std::find(fRunList.begin(), fRunList.end(), newPair) != fRunList.end()) {
415 std::cerr<<DYELLOW<<"Warning, adding run "<<std::setfill('0')<<std::setw(5)<<newPair.first<<"_"<<std::setw(3)<<newPair.second<<std::setfill(' ')<<" again!"<<RESET_COLOR<<std::endl;
416 return;
417 }
418 fRunList.push_back(newPair);
419
420 if(verbose) std::cout<<"adding run "<<runinfo->fRunNumber<<", sub run "<<runinfo->fSubRunNumber<<" to run "<<fRunNumber<<", sub run "<<fSubRunNumber<<std::endl;
421 // add the run length together
422 if(runinfo->fRunLength > 0) {
423 if(fRunLength > 0) {
424 fRunLength += runinfo->fRunLength;
425 } else {
426 fRunLength = runinfo->fRunLength;
427 }
428 }
429
430 if(runinfo->fRunNumber != fRunNumber) {
431 // check if the added run is an increment of the current run number (if the run number is set)
432 if(fRunNumber != 0) {
433 if(runinfo->fRunNumber + 1 == fRunNumber) {
434 if(verbose) std::cout<<"found second run ("<<runinfo->fRunNumber<<") before current run ("<<fRunNumber<<")"<<std::endl;
435 // use runinfo as first run and the current run as last
436 fFirstRunNumber = runinfo->fRunNumber;
437 fLastRunNumber = fRunNumber;
438 fRunStart = runinfo->fRunStart; // no need to set run stop
439 } else if(runinfo->fRunNumber - 1 == fRunNumber) {
440 if(verbose) std::cout<<"found second run ("<<runinfo->fRunNumber<<") after current run ("<<fRunNumber<<")"<<std::endl;
441 // use runinfo as last run and the current run as first
442 fFirstRunNumber = fRunNumber;
443 fLastRunNumber = runinfo->fRunNumber;
444 fRunStop = runinfo->fRunStop; // no need to set run start
445 } else {
446 if(verbose) std::cout<<"found second run ("<<runinfo->fRunNumber<<") non-consecutive to run ("<<fRunNumber<<")"<<std::endl;
447 // run start and stop don't make a lot of sense with non-consecutive runs (?)
448 fRunStart = 0.;
449 fRunStop = 0.;
450 // still need to keep some kind of information about the runs (e.g. to create filenames)
451 // by keeping first and last the exact same it is still obvious that these are not consecutive runs
452 fFirstRunNumber = fRunNumber;
453 fLastRunNumber = fRunNumber;
454 }
455 // the run number is meaningful only when the run numbers are the same
456 fRunNumber = 0;
457 fSubRunNumber = -1;
458 } else if(fFirstRunNumber != 0 && fLastRunNumber != 0 && fFirstRunNumber != fLastRunNumber) {
459 // if we already have a (good) range of runs, check if runinfo fits at the beginning or the end
460 if(runinfo->fRunNumber + 1 == fFirstRunNumber) {
461 if(verbose) std::cout<<"found another run ("<<runinfo->fRunNumber<<") before first run ("<<fFirstRunNumber<<")"<<std::endl;
462 // use runinfo as first run
463 fFirstRunNumber = runinfo->fRunNumber;
464 fRunStart = runinfo->fRunStart;
465 } else if(runinfo->fRunNumber - 1 == fLastRunNumber) {
466 if(verbose) std::cout<<"found another run ("<<runinfo->fRunNumber<<") after last run ("<<fLastRunNumber<<")"<<std::endl;
467 // use runinfo as last run
468 fLastRunNumber = runinfo->fRunNumber;
469 fRunStop = runinfo->fRunStop;
470 } else if(runinfo->fRunNumber == fFirstRunNumber || runinfo->fRunNumber == fLastRunNumber) {
471 if(verbose) std::cout<<"found another sub(?) run part of runs ("<<fFirstRunNumber<<" - "<<fLastRunNumber<<")"<<std::endl;
472 // found probably another subrun of a run already added
473 // since we do not keep track of all subruns we have to assume this is in order
474 // so we only update the run start or stop if necessary
475 if(verbose) std::cout<<"changing run start/stop from "<<std::setw(16)<<fRunStart<<"/"<<std::setw(16)<<fRunStop<<" to ";
476 if(fRunStop < runinfo->fRunStop) fRunStop = runinfo->fRunStop;
477 if(fRunStart > runinfo->fRunStart) fRunStart = runinfo->fRunStart;
478 if(verbose) std::cout<<std::setw(16)<<fRunStart<<"/"<<std::setw(16)<<fRunStop<<std::endl;
479 } else {
480 if(verbose) std::cout<<"found another run ("<<runinfo->fRunNumber<<") non-consecutive to runs ("<<fFirstRunNumber<<" - "<<fLastRunNumber<<")"<<std::endl;
481 // run start and stop don't make a lot of sense with non-consecutive runs (?)
482 fRunStart = 0.;
483 fRunStop = 0.;
484 // still need to keep some kind of information about the runs (e.g. to create filenames)
485 // by keeping first and last the exact same it is still obvious that these are not consecutive runs
486 fLastRunNumber = fFirstRunNumber;
487 }
488 } else {
489 // the run number is zero, and we do not have a (good) range, so there is nothing to do.
490 if(verbose) std::cout<<"found another run ("<<runinfo->fRunNumber<<") non-consecutive run ("<<fFirstRunNumber<<" - "<<fLastRunNumber<<")"<<std::endl;
491 }
492 } else if(fSubRunNumber != -1) {
493 // check if the added sub run is an increment of the current run number
494 if(runinfo->fSubRunNumber + 1 == fSubRunNumber) {
495 if(verbose) std::cout<<"found second sub run ("<<runinfo->fSubRunNumber<<") before current sub run ("<<fSubRunNumber<<")"<<std::endl;
496 // if the run numbers are the same and we have subsequent sub runs we can update the run start
497 fRunStart = runinfo->fRunStart;
498 fFirstSubRunNumber = runinfo->fSubRunNumber;
499 fLastSubRunNumber = fSubRunNumber;
500 } else if(runinfo->fSubRunNumber - 1 == fSubRunNumber) {
501 if(verbose) std::cout<<"found second sub run ("<<runinfo->fSubRunNumber<<") after current sub run ("<<fSubRunNumber<<")"<<std::endl;
502 // if the run numbers are the same and we have subsequent sub runs we can update the run stop
503 fRunStop = runinfo->fRunStop;
504 fFirstSubRunNumber = fSubRunNumber;
505 fLastSubRunNumber = runinfo->fSubRunNumber;
506 } else {
507 if(verbose) std::cout<<"found second sub run ("<<runinfo->fSubRunNumber<<") non-consecutive to current sub run ("<<fSubRunNumber<<")"<<std::endl;
508 // with multiple non-sequential subruns added, the sub run number and start/stop have no meaning anymore
509 fRunStart = 0.;
510 fRunStop = 0.;
511 }
512 // sub run number is only meaningful if it's the only sub run
513 fSubRunNumber = -1;
514 } else {
515 // we have the same run with a range of sub-runs already added, so check if this once fits at the end or the beginning
516 if(runinfo->fSubRunNumber + 1 == fFirstSubRunNumber) {
517 if(verbose) std::cout<<"found another sub run ("<<runinfo->fSubRunNumber<<") before first sub run ("<<fFirstSubRunNumber<<")"<<std::endl;
518 // use runinfo as first run
519 fFirstSubRunNumber = runinfo->fSubRunNumber;
520 fRunStart = runinfo->fRunStart;
521 } else if(runinfo->fSubRunNumber - 1 == fLastSubRunNumber) {
522 if(verbose) std::cout<<"found another sub run ("<<runinfo->fSubRunNumber<<") after last sub run ("<<fLastSubRunNumber<<")"<<std::endl;
523 // use runinfo as last run
524 fLastSubRunNumber = runinfo->fSubRunNumber;
525 fRunStop = runinfo->fRunStop;
526 } else {
527 if(verbose) std::cout<<"found another sub run ("<<runinfo->fSubRunNumber<<") non-consecutive to sub runs ("<<fFirstSubRunNumber<<" - "<<fLastSubRunNumber<<")"<<std::endl;
528 // with multiple non-sequential subruns added, the sub run number and start/stop have no meaning anymore
529 fRunStart = 0.;
530 fRunStop = 0.;
531 fFirstSubRunNumber = -1;
532 fLastSubRunNumber = -1;
533 }
534 }

550std::string TRunInfo::CreateLabel(bool quiet)
551{
552 /// This function creates a label/string based on the run number and the subrun number.
553 auto runInfo = Get();
554 Int_t runNumber = runInfo->RunNumber();
555 Int_t subRunNumber = runInfo->SubRunNumber();
556
557 std::string result;
558 if(!quiet) std::cout<<"Using run number "<<runNumber<<", sub run number "<<subRunNumber<<", first/last run number "<<runInfo->FirstRunNumber()<<"/"<<runInfo->LastRunNumber()<<", and first/last sub run number"<<runInfo->FirstSubRunNumber()<<"/"<<runInfo->LastSubRunNumber()<<std::endl;
559 if(runNumber != 0 && subRunNumber != -1) {
560 // both run and subrun number set => single file processed
561 result = Form("%05d_%03d", runNumber, subRunNumber);
562 } else if(runNumber != 0) {
563 // multiple subruns of a single run
564 // we could check if first and last sub run number are both -1 (which is non-consecutive runs or not initialized, the latter would mean single file w/o a subrun number, like ILL data)
565 result = Form("%05d_%03d-%03d", runNumber, runInfo->FirstSubRunNumber(), runInfo->LastSubRunNumber());
566 } else {
567 // multiple runs
568 result = Form("%05d-%05d", runInfo->FirstRunNumber(), runInfo->LastRunNumber());
569 }
570 if(!quiet) std::cout<<"Created label "<<result<<std::endl;
571
572 return result;
