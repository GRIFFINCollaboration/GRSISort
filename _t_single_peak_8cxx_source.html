76Double_t TSinglePeak::FWHM()
77{
78 /// Return the full width at half-maximum.
79 if(fPeakFunction == nullptr) {
80 std::cerr<<__PRETTY_FUNCTION__<<": peak function ("<<fPeakFunction<<") is null"<<std::endl;
81 return 0.;
82 }
83 UpdatePeakParameters();
84 double low = Centroid() - 10.*Sigma();
85 double high = Centroid() + 10.*Sigma();
86 auto maxX = fPeakFunction->GetMaximumX(low, high);
87 auto max = fPeakFunction->Eval(maxX);
88 return (fPeakFunction->GetX(max/2., maxX, maxX+10.*Sigma()) - fPeakFunction->GetX(max/2., maxX-10.*Sigma(), maxX));
89}
