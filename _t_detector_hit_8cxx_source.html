69Float_t TDetectorHit::GetCharge() const
70{
71 TChannel* channel = GetChannel();
72 if(channel == nullptr) {
73 return Charge();
74 }
75 if(fKValue > 0 && !channel->UseCalFileIntegration()) {
76 return Charge() / (static_cast<Float_t>(fKValue)); // this will use the integration value
77 }
78 if(channel->UseCalFileIntegration()) {
79 return Charge() / (static_cast<Float_t>(channel->GetIntegration())); // this will use the integration value
80 } // in the TChannel if it exists.

84double TDetectorHit::GetEnergy(Option_t*) const
85{
86 if(TestHitBit(EBitFlag::kIsEnergySet)) {
87 return fEnergy;
88 }
89 TChannel* channel = GetChannel();
90 if(channel == nullptr) {
91 return SetEnergy(static_cast<Double_t>(Charge()));
92 }
93 if(channel->UseCalFileIntegration()) {
94 double energy = channel->CalibrateENG(Charge(), 0);
95 return SetEnergy(energy +
96 GetEnergyNonlinearity(energy)); // this will use the integration value
97 // in the TChannel if it exists.
98 }
99 if(fKValue > 0) {
100 double energy = channel->CalibrateENG(Charge(), static_cast<int>(fKValue));
101 return SetEnergy(energy + GetEnergyNonlinearity(energy));
102 }
103 double energy = channel->CalibrateENG(Charge());
104 return SetEnergy(energy + GetEnergyNonlinearity(energy));
105}
