51TXMLNode* TXMLOdb::FindNode(const char* name, TXMLNode* node)
52{
53 /// Finds node with name "name". If a node is provided this node will be used as a starting point.
54 /// If the provided node is a null pointer fOdb is used instead. Returns a null pointer if the
55 /// search fails.
56 if(node == nullptr) {
57 if(fOdb == nullptr) {
58 return nullptr;
59 }
60 node = fOdb; //->GetChildren();
61 }
62 if(!node->HasChildren()) {
63 return nullptr;
64 }
65 node = node->GetChildren();
66 while(node != nullptr) {
67 std::string nodename = GetNodeName(node);
68 if(nodename.compare(name) == 0) {
69 return node;
70 }
71 node = node->GetNextNode();
72 }
73
74 return nullptr;
