48 static TGRSIProof* Open(const char* worker = "")
49 {
50 TGRSIProof* p = static_cast<TGRSIProof*>(TProof::Open(worker));
51 if(p != nullptr) p->LoadLibsIntoProof();
52 return p;
53 }
54 void LoadLibsIntoProof()
55 {
56 // This function Loops through all libraries in GRSISort, and brute-force loads them into proof
57 // This is done because order of loading matters. We suppress the cout output because proof->Exec
58 // barfs all of the output. To me there is no obvious way to load every library on each of the PROOF slaves
59
60 const char* pPath = getenv("GRSISYS");
61
62 // First set the include path on each slave
63 Exec(Form(R"(gInterpreter->AddIncludePath("%s/include"))", pPath));
64 // if we have a data parser/detector library load it
65 std::string library = TGRSIOptions::Get()->ParserLibrary();
66 if(library.empty()) {
67 std::cout << DYELLOW << "Not loading any parser library, this might not work!" << RESET_COLOR << std::endl;
68 std::cout << "Loading library '" << pPath << "/lib/libGRSI.so'" << std::endl;
69 Exec(Form(R"(gSystem->Load("%s/lib/libGRSI.so");)", pPath));
70 } else {
71 std::cout << "Loading parser library '" << library << "'" << std::endl;
72 Exec(Form(R"(gSystem->Load("%s");)", library.c_str()));
73 TParserLibrary::Get()->Load();
74 }
75 }
76
77 /// \cond CLASSIMP
78 ClassDefOverride(TGRSIProof, 1); // Event Fragments
79 /// \endcond
80};
