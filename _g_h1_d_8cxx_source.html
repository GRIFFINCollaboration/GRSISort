213GH1D* GH1D::Project(int bins)
214{
215 GH1D* proj = nullptr;
216 double ymax = GetMinimum();
217 double ymin = GetMaximum();
218 if(bins == -1) {
219 bins = static_cast<int>(std::abs(ymax - ymin));
220 if(bins < 1) {
221 bins = 100;
222 }
223 }
224 proj = new GH1D(Form("%s_y_axis_projection", GetName()), Form("%s_y_axis_projection", GetName()), bins, ymin, ymax);
225 for(int i = 0; i < GetNbinsX(); i++) {
226 if(GetBinContent(i) != 0) {
227 proj->Fill(GetBinContent(i));
228 }
229 }
230
231 return proj;
