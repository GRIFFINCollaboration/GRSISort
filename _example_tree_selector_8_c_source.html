54 // we could check multiplicities here and skip events where we do not have at least
55 // three suppressed addback energies and a beta, but we want to fill some general
56 // histograms without these cuts.

58 // clear the vectors and other variables

64
65 //Loop over all suppressed addback Griffin Hits

67 auto grif1 = fGrif->GetSuppressedAddbackHit(i);
68 fH1.at("asE")->Fill(grif1->GetEnergy());
69 fSuppressedAddback[0] = grif1->GetEnergy();
70 for(auto j = i+1; j < fGrif->GetSuppressedAddbackMultiplicity(fGriffinBgo); ++j) {
71 auto grif2 = fGrif->GetSuppressedAddbackHit(j);
72 fSuppressedAddback[1] = grif2->GetEnergy();
73 for(auto k = j+1; k < fGrif->GetSuppressedAddbackMultiplicity(fGriffinBgo); ++k) {
74 auto grif3 = fGrif->GetSuppressedAddbackHit(k);
75 fSuppressedAddback[2] = grif3->GetEnergy();
76 // we now have three suppressed addback hits i, j, and k so now we need a coincident beta-tag
77 bool foundBeta = false;
78 for(auto b = 0; b < fZds->GetMultiplicity(); ++b) {
79 auto beta = fZds->GetZeroDegreeHit(b);
80 if(b == 0) {
