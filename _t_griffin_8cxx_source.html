425Short_t TGriffin::GetAddbackMultiplicity(const EGainBits& gain_type)
426{
427 // Automatically builds the addback hits using the fAddbackCriterion (if the size of the fAddbackHits vector is zero)
428 // and return the number of addback hits.
429 if(!IsCrossTalkSet(gain_type)) {
430 // Calculate Cross Talk on each hit
431 FixCrossTalk(gain_type);
432 }
433 auto& hit_vec = GetHitVector(gain_type);
434 auto& ab_vec = GetAddbackVector(gain_type);
435 auto& frag_vec = GetAddbackFragVector(gain_type);
436 if(hit_vec.empty()) {
437 return 0;
438 }
439 // if the addback has been reset, clear the addback hits
440 if(!IsAddbackSet(gain_type)) {
441 for(auto& hit : ab_vec) {
442 delete hit;
443 }
444 ab_vec.clear();
445 frag_vec.clear();
446 }
447 if(ab_vec.empty()) {
448 CreateAddback(hit_vec, ab_vec, frag_vec);
449 SetAddback(gain_type, true);
450 }
451
452 return ab_vec.size();
453}

475void TGriffin::AddFragment(const std::shared_ptr<const TFragment>& frag, TChannel* chan)
476{
477 // Builds the GRIFFIN Hits directly from the TFragment. Basically, loops through the hits for an event and sets
478 // observables.
479 // This is done for both GRIFFIN and it's suppressors.
480 if(frag == nullptr || chan == nullptr) {
481 return;
482 }
483 const TMnemonic* mnemonic = chan->GetMnemonic();
484 if(mnemonic == nullptr) {
485 std::cerr<<"Trying to add fragment to TGriffin w/o mnemonic in TChannel!"<<std::endl;
486 return;
487 }
488
489 if(mnemonic->SubSystem() != TMnemonic::EMnemonic::kG) {
490 std::cerr<<__PRETTY_FUNCTION__<<": not a GRIFFIN detector: "<<static_cast<std::underlying_type<TMnemonic::EMnemonic>::type>(mnemonic->SubSystem())<<std::endl;
491 return;
492 }
493
494 // only create the hit if we have a HPGe signal (A or B), not a suppressor (S)
495 switch(mnemonic->OutputSensor()) {
496 case TMnemonic::EMnemonic::kA:
497 {
498 TGriffinHit* hit = new TGriffinHit(*frag);
499 GetHitVector(EGainBits::kLowGain).push_back(hit);
500 }
501 break;
502 case TMnemonic::EMnemonic::kB:
503 {
504 TGriffinHit* hit = new TGriffinHit(*frag);
505 GetHitVector(EGainBits::kHighGain).push_back(hit);
506 }
507 break;
508 default:
509 std::cout<<"output sensor "<<static_cast<std::underlying_type<TMnemonic::EMnemonic>::type>(mnemonic->OutputSensor())<<", skipping it"<<std::endl;
510 break;
511 };
512}

514TVector3 TGriffin::GetPosition(int DetNbr, int CryNbr, double dist)
515{
516 // Gets the position vector for a crystal specified by CryNbr within Clover DetNbr at a distance of dist mm away.
517 // This is calculated to the most likely interaction point within the crystal.
518 if(DetNbr > 16) {
519 return TVector3(0, 0, 1);
520 }
521
522 TVector3 temp_pos(gCloverPosition[DetNbr]);
523
524 // Interaction points may eventually be set externally. May make these members of each crystal, or pass from
525 // waveforms.
526 Double_t cp = 26.0; // Crystal Center Point mm.
527 Double_t id = 45.0; // 45.0; //Crystal interaction depth mm.
528 // Set Theta's of the center of each DETECTOR face
529 ////Define one Detector position
530 TVector3 shift;
531 switch(CryNbr) {
532 case 0: shift.SetXYZ(-cp, cp, id); break;

608Double_t TGriffin::CTCorrectedEnergy(const TGriffinHit* const hit_to_correct, const TGriffinHit* const other_hit,
609 Bool_t time_constraint)
610{
611 if((hit_to_correct == nullptr) || (other_hit == nullptr)) {
612 std::cerr<<"One of the hits is invalid in TGriffin::CTCorrectedEnergy"<<std::endl;
613 return 0;
614 }
615
616 if(time_constraint) {
617 // Figure out if this passes the selected window
618 if(TMath::Abs(other_hit->GetTime() - hit_to_correct->GetTime()) >
619 TGRSIOptions::AnalysisOptions()->AddbackWindow()) { // placeholder
620 return hit_to_correct->GetEnergy();
621 }
622 }
623
624 if(hit_to_correct->GetDetector() != other_hit->GetDetector()) {
625 return hit_to_correct->GetEnergy();
626 }
627 static bool been_warned[256] = {false};
628 double fixed_energy = hit_to_correct->GetEnergy();
629 try {
630 if(hit_to_correct->GetChannel() != nullptr) {
631 fixed_energy -= hit_to_correct->GetChannel()->GetCTCoeff().at(other_hit->GetCrystal()) * other_hit->GetNoCTEnergy();
632 }
633 } catch(const std::out_of_range& oor) {
634 int id = 16 * hit_to_correct->GetDetector() + 4 * hit_to_correct->GetCrystal() + other_hit->GetCrystal();
635 if(!been_warned[id]) {
636 been_warned[id] = true;
637 std::cerr<<DRED<<"Missing CT correction for Det: "<<hit_to_correct->GetDetector()
638 <<" Crystals: "<<hit_to_correct->GetCrystal()<<" "<<other_hit->GetCrystal()<<" (id "<<id<<")"<<RESET_COLOR<<std::endl;
639 }
640 return hit_to_correct->GetEnergy();
641 }
642
643 return fixed_energy;
644}

656void TGriffin::FixCrossTalk(const EGainBits& gain_type)
657{
658 if(!TGRSIOptions::AnalysisOptions()->IsCorrectingCrossTalk()) return;
659
660 auto& hit_vec = GetHitVector(gain_type);
661 if(hit_vec.size() < 2) {
662 SetCrossTalk(gain_type, true);
663 return;
664 }
665 for(auto& hit : hit_vec) {
666 static_cast<TGriffinHit*>(hit)->ClearEnergy();
667 }
668
669 for(auto& one : hit_vec) {
670 for(auto& two : hit_vec) {
671 one->SetEnergy(CTCorrectedEnergy(static_cast<TGriffinHit*>(one), static_cast<TGriffinHit*>(two)));
672 }
673 }
674 SetCrossTalk(gain_type, true);
675}
