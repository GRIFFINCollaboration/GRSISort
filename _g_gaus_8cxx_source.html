96bool GGaus::InitParams(TH1* fithist)
97{
98 if(fithist == nullptr) {
99 std::cout<<"No histogram is associated yet, no initial guesses made"<<std::endl;
100 return false;
101 }
102 // Makes initial guesses at parameters for the fit. Uses the histogram to
103 Double_t xlow, xhigh;
104 GetRange(xlow, xhigh);
105
106 Int_t binlow = fithist->GetXaxis()->FindBin(xlow);
107 Int_t binhigh = fithist->GetXaxis()->FindBin(xhigh);
108
109 Double_t highy = fithist->GetBinContent(binlow);
110 Double_t lowy = fithist->GetBinContent(binhigh);
111 for(int x = 1; x < 5; x++) {
112 highy += fithist->GetBinContent(binlow - x);
113 lowy += fithist->GetBinContent(binhigh + x);
114 }
115 highy = highy / 5.0;
116 lowy = lowy / 5.0;
117
118 if(lowy > highy) {
119 std::swap(lowy, highy);
120 }
121
122 double largestx = 0.0;
123 double largesty = 0.0;
124 int i = binlow;
125 for(; i <= binhigh; i++) {
126 if(fithist->GetBinContent(i) > largesty) {
127 largesty = fithist->GetBinContent(i);
128 largestx = fithist->GetXaxis()->GetBinCenter(i);
129 }
130 }
131
132 // - par[0]: height of peak
133 // - par[1]: cent of peak
134 // - par[2]: sigma
135 // - par[3]: bg constent
136 // - par[4]: bg slope
137
138 // limits.
139 TF1::SetParLimits(0, 0, largesty * 2);
140 TF1::SetParLimits(1, xlow, xhigh);
141 TF1::SetParLimits(2, 0, xhigh - xlow);
142
143 // Make initial guesses
144 TF1::SetParameter(0, largesty); // fithist->GetBinContent(bin));
145 TF1::SetParameter(1, largestx); // GetParameter("centroid"));
146 TF1::SetParameter(2, (largestx * .01) / 2.35); // 2,(xhigh-xlow)); //2.0/binWidth); //
147
148 TF1::SetParError(0, 0.10 * largesty);
149 TF1::SetParError(1, 0.25);
150 TF1::SetParError(2, 0.10 * ((largestx * .01) / 2.35));
151
152 SetInitialized();
153 return true;
154}

156Bool_t GGaus::Fit(TH1* fithist, Option_t* opt)
157{
158 if(fithist == nullptr) {
159 return false;
160 }
161 TString options = opt;
162 if(!IsInitialized()) {
163 InitParams(fithist);
164 }
165 TVirtualFitter::SetMaxIterations(100000);
166
167 bool verbose = !options.Contains("Q");
168 bool noprint = options.Contains("no-print");
169 if(noprint) {
170 options.ReplaceAll("no-print", "");
171 }
172
173 if(fithist->GetSumw2()->fN != fithist->GetNbinsX() + 2) {
174 fithist->Sumw2();
175 }
176
177 TFitResultPtr fitres = fithist->Fit(this, Form("%sRSME", options.Data()));
178
179 if(!fitres.Get()->IsValid()) {
180 if(!verbose) {
181 std::cout<<RED<<"fit has failed, trying refit... "<<RESET_COLOR;
182 }
183 fithist->GetListOfFunctions()->Last()->Delete();
184 fitres = fithist->Fit(this, Form("%sRSME", options.Data())); //,Form("%sRSM",options.Data()))
185 if(fitres.Get()->IsValid()) {
186 if(!verbose && !noprint) {
187 std::cout<<DGREEN<<" refit passed!"<<RESET_COLOR<<std::endl;
188 }
189 } else {
190 if(!verbose && !noprint) {
191 std::cout<<DRED<<" refit also failed :( "<<RESET_COLOR<<std::endl;
192 }
193 }
194 }
195
196 Double_t xlow, xhigh;
197 TF1::GetRange(xlow, xhigh);
198
199 double bgpars[2];
200 bgpars[0] = TF1::GetParameters()[3];
201 bgpars[1] = TF1::GetParameters()[4];
202
203 fBGFit.SetParameters(bgpars);
204
205 fArea = Integral(xlow, xhigh) / fithist->GetBinWidth(1);
206 double bgArea = fBGFit.Integral(xlow, xhigh) / fithist->GetBinWidth(1);
207 ;
208 fArea -= bgArea;
209
210 if(xlow > xhigh) {
211 std::swap(xlow, xhigh);
212 }
213 fSum = fithist->Integral(fithist->GetXaxis()->FindBin(xlow),
214 fithist->GetXaxis()->FindBin(xhigh)); //* fithist->GetBinWidth(1);
215 std::cout<<"sum between markers: "<<fSum<<std::endl;
216 fDSum = TMath::Sqrt(fSum);
217 fSum -= bgArea;
218 std::cout<<"sum after subtraction: "<<fSum<<std::endl;
219
220 if(!verbose && !noprint) {
221 Print();
222 }
223
224 Copy(*fithist->GetListOfFunctions()->FindObject(GetName()));
225 fithist->GetListOfFunctions()->Add(fBGFit.Clone());
226
227 return true;

263void GGaus::DrawResiduals(TH1* hist) const
264{
265 if(hist == nullptr) {
266 return;
267 }

283 bin[points] = hist->GetBinCenter(i);
284 points++;
285 }
286 new GCanvas();
287 auto* residuals = new TGraph(points, bin, res);
288 residuals->Draw("*AC");
